    class ThickClientApplicationHandler(object):
    """
    Description:
        |  This class contains methods related to thick client handler . Initiation of handler is handled here

    """

    str_cts_title = "Content Tool Suite"

    def __init__(self, thick_client_config):

        self.str_app_path = thick_client_config.get("app_path")
        self.str_processname = thick_client_config.get("process_name")
        self.str_application = thick_client_config.get("application")
        self.bln_connect_to_open_app = thick_client_config.get("connect_to_open_app")
        self.bln_thickclient_teardown_flag = thick_client_config.get("thickclient_teardown_flag")
        self.bln_reset_objects_flag = thick_client_config.get("reset_objects")
        self.__exceptions_generic = Exceptions(os.getcwd()).Generic()
        self.__exceptions_thickclient = Exceptions(os.getcwd()).Thick_Client()
        self.logger = LoggerClass().logger

        if self.bln_thickclient_teardown_flag is False:
            self.bln_connect_to_open_app = False

    def get_handler(self):
        """
        Description:
            |  This method will return a thick client handler based on the application name passed while initiating the thickclientHandlerFactory class .

        :return: thick client handler

        """
        try:
            if (self.str_application == "cts"):
                return self.launch_CTS()
            elif (self.str_application == "excel"):
                return self.launch_excel()
            elif (self.str_application == "extractor"):
                return self.launch_extractor()
            elif (self.str_application == "celsus"):
                return self.launch_celsus()
            elif (self.str_application == "celsusprod"):
                return self.launch_ciq_content_collection_applications("Celsus.*")
            elif (self.str_application == 'webcrawlerentity'):
                return self.launch_content_crawler_entity()
            elif (self.str_application == 'nonportfolio'):
                return self.launch_ciq_content_collection_applications("NonPortfolio Collection.*")
            elif (self.str_application == 'keydevcollection'):
                return self.launch_ciq_content_collection_applications("KeyDev Collection.*")
            else:
                return self.launch(self.str_app_path)
        except Exception as e:
            self.__exceptions_generic.raise_generic_exception(
                pstr_message=str(e),
                pbool_insert_kibana=True,
                pbool_trim_log=True, pbool_log_local_exceptions=True,
                pbool_fail_on_exception=True, pbool_use_pytest_format=False)

    def launch_CTS(self):
        """
        Description:
        	|  This method allows user to launch CTS application.

        Examples:
        	|  launch_CTS()

        .. note::
        	|  Internally this method use "bln_connect_to_open_app" parameter if user wants to connect existing opened CTS application.
        	|  If "bln_connect_to_open_app" parameter is set as True then it tries to connect with existing open CTS application. This parameter is useful when we debug the code.
        	|  If "bln_connect_to_open_app" parameter is set as False then first check if CTS application is open then this method will close it and then it will launch CTS application.

        """

        try:
            if (not self.bln_connect_to_open_app):
                # Kill existing running CTS application
                for proc in psutil.process_iter():
                    try:
                        # Get process name & pid from process object.
                        if proc.name().lower().__contains__(self.str_processname.lower()):
                            proc.kill()

                    except Exception as e:
                        self.__exceptions_generic.raise_generic_exception(
                            pstr_message='Error occurred while killing the given process -->' + str(e),
                            pbool_insert_kibana=True,
                            pbool_trim_log=True,
                            pbool_log_local_exceptions=True,
                            pbool_fail_on_exception=True, pbool_use_pytest_format=False)
                if (self.bln_reset_objects_flag == True and self.bln_connect_to_open_app == False):
                    self.reset_objects()

                # Launch CTS application
                self.launch(self.str_app_path)
                app = self.connect_to_app(pstr_title_re=self.str_cts_title, pstr_control_type="Window",
                                          pint_timeout=300)
                app.window(title_re=self.str_cts_title, control_type="Window").maximize()
                return app

            elif (self.bln_connect_to_open_app):
                app = self.connect_to_app(pstr_title_re=self.str_cts_title, pstr_control_type="Window", pint_timeout=60)
                app.window(title_re=self.str_cts_title, control_type="Window").maximize()
                return app
            else:
                self.__exceptions_generic.raise_generic_exception(
                    pstr_message="Provide connect_to_open_app parameter value as either True or False",
                    pbool_insert_kibana=True,
                    pbool_trim_log=True, pbool_log_local_exceptions=True,
                    pbool_fail_on_exception=True, pbool_use_pytest_format=False)

        except Exception as e:
            self.__exceptions_generic.raise_generic_exception(
                pstr_message=str(e),
                pbool_insert_kibana=True,
                pbool_trim_log=True, pbool_log_local_exceptions=True,
                pbool_fail_on_exception=True, pbool_use_pytest_format=False)
