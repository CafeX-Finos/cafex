    def click(self, locator, explicit_wait=None):
        """
        Description:
        |  This method will click on the locator provided . It will find the element for the locator given , convert it into a web element then wait for it to be clickable , the time of wait will depend on the value passed in the explicit wait

        :param locator: This parameter will contain the locator in a fixed format which is , locatortype=locator . For eg. id=username or xpath=.//*[@id='username']
        :type locator: String
        :param explicit_wait: This parameter will by default be none and the number passed in it would be the explicit wait(in seconds) for the particular element
        :type explicit_wait: Integer


        :return: None

        Examples:
            |  PyGlobals().web_client_actions.click("xpath=//a[@class='advanced-search-link']")
            |  PyGlobals().web_client_actions.click("xpath=//a[@class='advanced-search-link']",30)

        """
        try:
            self.get_clickable_web_element(locator, explicit_wait).click()
        except Exception as e:
            custom_exception_message = "Exception in click method for locator: " + str(
                locator) + " .Exception Details: " + str(repr(e))
            self.__exceptions_generic.raise_generic_exception(pstr_message=str(custom_exception_message),
                                                              pbool_insert_kibana=True,
                                                              pbool_trim_log=True, pbool_log_local_exceptions=True,
                                                              pbool_fail_on_exception=False,
                                                              pbool_use_pytest_format=True)
            raise

    def type(self, locator, text, pbool_clear=False, explicit_wait=None, pbool_click_before_type=True):
        """
        Description:
        |  This method will type the given value on the locator provided . It will find the element for the locator given , convert it into a web element then wait for it to be clickable , the time of wait will depend on the value passed in the explicit wait . It will then click on the element and finally type in the value

        :param locator: This parameter will contain the locator in a fixed format which is , locatortype=locator . For eg. id=username or xpath=.//*[@id='username']
        :type locator: String
        :param text: This parameter will contain the value to be typed into the web element
        :type text: String
        :param explicit_wait: This parameter will by default be none and the number passed in it would be the explicit wait(in seconds) for the particular element
        :type explicit_wait: Integer
        :param pbool_clear: This parameter is to use clear feature,if this is set to True respective field content will be cleared,By default this is set to False and pbool_clear only accepts to boolean value
        :type pbool_clear: Boolean
        :param pbool_click_before_type: This parameter is used to determine whether there will be a click on the element or not before typing the given text
        :type pbool_click_before_type: Boolean


        :return: None

        Examples:
            |  PyGlobals().web_client_actions.type("xpath=//a[@class='username']","your_username")
            |  PyGlobals().web_client_actions.type("xpath=//a[@class='username']","your_username",30)
        """

            if pbool_click_before_type:
                self.get_clickable_web_element(locator, explicit_wait).click()
            if pbool_clear:
                self.get_clickable_web_element(locator, explicit_wait).clear()

        except Exception as e:
            custom_exception_message = "Exception in type method for locator: " + str(locator) + " and text: " + str(
                text) + ".Exception Details: " + str(repr(e))

    def is_element_present(self, locator, explicit_wait=None):
        """
        Description:
        |  This method will verify if the given locator is present on the page or not for the given time and return True/False based on the result.
        |  If the wait time need to be updated for a particular element , updated time can be passed in parameters which will work only for this element

        :param locator: This parameter will contain the locator in a fixed format which is , locatortype=locator . For eg. id=username or xpath=.//*[@id='username']
        :type locator: String - Locator can only be xpath/id/css/classname
        :param explicit_wait: This parameter will by default be the default explicit wait time in configuration file and the number passed in it would be the explicit wait(in seconds) for the particular element
        :type explicit_wait: Integer

        :return: Bool

        Examples:
            |  PyGlobals().web_client_actions.is_element_present("xpath=//*[@name='password']")

        .. notes::
            |  If no value is passed in for the explicit wait it will by use the default explicit wait from the configuration file.
        """
        try:

            if explicit_wait is None:
                explicit_wait = self.__get_explicit_wait()

            if isinstance(locator, str):
                locator_details = locator.split('=', 1)

                if locator.lower().find('accessibility id') != -1:
                    locator = locator.replace("accessibility id", "id")

                element = WebDriverWait(self.driver, explicit_wait).until(
                    EC.presence_of_element_located((self.get_locator_strategy(locator_details[0]), locator_details[1])))
                if element is not None:
                    return True
                else:
                    return False
            elif isinstance(locator, WebElement):
                return locator.is_displayed()
        except Exception as e:
            custom_exception_message = "Exception in is_element_present method for locator: " + str(
                locator) + ".Exception Details: " + str(repr(e))
            self.__exceptions_generic.raise_generic_exception(pstr_message=str(custom_exception_message),
                                                              pbool_insert_kibana=False,
                                                              pbool_trim_log=True, pbool_log_local_exceptions=True,
                                                              pbool_fail_on_exception=False,
                                                              pbool_use_pytest_format=True)
            return False

    def get_xpath(self, **kwargs):
        """
        Description:
        	|  This method will create an xpath for the given tag/element type , attribute and its value

        :param tag: This is the element type . For eg : a or input or div
        :type tag: String
        :param attribute: This is the name of the attribute. For eg:value
        :type attribute: String
        :param value: The is the value of the attribute
        :type value: String

        :return: String
        Examples:
        	|  PyGlobals().web_client_actions.get_xpath(tag='a',attribute='text',value='new')
        	|  The above line will return
        	|  .//a[text='new']

        """
        try:
            if 'tag' not in kwargs:
                tag = "*"
            else:
                tag = kwargs.get("tag")

            if 'attribute' not in kwargs:
                return ".//" + tag
            else:
                attribute = kwargs.get("attribute")
                if 'value' not in kwargs:
                    return ".//" + tag + "[" + attribute + "=" + "''" + "]"
                else:
                    return ".//" + tag + "[" + attribute + "=" + "'" + kwargs.get("value") + "'" + "]"
        except Exception as e:
            custom_exception_message = "Exception in get_xpath method .Exception Details: " + str(repr(e))
            self.__exceptions_generic.raise_generic_exception(pstr_message=str(custom_exception_message),
                                                              pbool_insert_kibana=True,

    def scroll(self, scroll_type, **kwargs):
        """
        Description:
        	|  This method is used to scroll on web pages

        :param scroll_type: User can pass the type of scroll, one of the four values - height,bottom,infinite,web_element
        :type scroll_type: String
        :param value(optional): Determined the length of scroll
        :type value(optional): Integer
        :param scroll_pause_time: Time to wait(in secs) before scrolling to the end of page
        :type scroll_pause_time: Integer
        :param element(optional): Used to pass web element when scroll_type is "web_element"
        :type element(optional): WebElement
        :param locator(optional): Used to pass locator of web element when scroll_type is "web_element"
        :type locator(optional): String

        :return: None
        Examples:
        	|  PyGlobals().web_client_actions.scroll("height",value=50) -  to scroll to a particular height
        	|  PyGlobals().web_client_actions.scroll("bottom") - to scroll to a end of current page , in case where the page length is limited and bottom of page is the bottom of the current page loaded
        	|  PyGlobals().web_client_actions.scroll("infinite",scroll_pause_time=1) , this is used in cases where page has ajax calls and page loads on reaching the bottom of the current page. scroll_pause_time is the number of seconds user want to wait before scrolling again
        	|  PyGlobals().web_client_actions.scroll("web_element",locator="xpath=.//[id='password']") , used to scroll to the web element of the given locator
        	|  PyGlobals().web_client_actions.scroll("web_element",element=web_element) , used to scroll to the given web element


        """
        try:
            if scroll_type == 'height':
                if 'value' in kwargs:
                    self.execute_javascript("window.scrollTo(0, arguments[0])", kwargs.get('value'))
                else:
                    raise Exception("Height to be scrolled not given")
            elif scroll_type == 'bottom':
                self.execute_javascript("window.scrollTo(0, document.body.scrollHeight);")
            elif scroll_type == 'infinite':
                count = 0
                if 'value' in kwargs:
                    scroll_pause_time = kwargs.get("scroll_pause_time")
                else:
                    scroll_pause_time = 1
                last_height = self.execute_javascript("return document.body.scrollHeight")
                while True and count < 20:
                    count = count + 1
                    self.execute_javascript("window.scrollTo(0, document.body.scrollHeight);")

                    time.sleep(scroll_pause_time)

                    new_height = self.execute_javascript("return document.body.scrollHeight")
                    if new_height == last_height:
                        break
                    last_height = new_height
            elif scroll_type == 'web_element':
                if 'element' in kwargs:
                    self.execute_javascript('arguments[0].scrollIntoView(true);', kwargs.get('element'))
                elif 'locator' in kwargs:
                    web_element = self.get_web_element(kwargs.get('locator'))
                    self.execute_javascript('arguments[0].scrollIntoView(true);', web_element)
                else:
                    raise Exception('Element or locator not passed where driver needs to scroll')
            else:
                raise Exception('Incorrect Scroll type , the scroll type can be height/bottom/infinite/web_element')
        except Exception as e:
            custom_exception_message = "Exception in scroll method .Exception Details: " + str(repr(e))
            self.__exceptions_generic.raise_generic_exception(pstr_message=str(custom_exception_message),
                                                              pbool_insert_kibana=True,
                                                              pbool_trim_log=True, pbool_log_local_exceptions=True,
                                                              pbool_fail_on_exception=False,
                                                              pbool_use_pytest_format=True)
            raise

    def search_broken_links(self):
        """
        Description:
            |  A simple function to fetch the total links on the web page and let the
            |  user know how many links on the function are working fine and which are broken.
            |

        :param search_broken_links:
        :type search_broken_links: None

        :return: int (no. of broken links in a page)

        Examples:
            |  PyGlobals().business_utils_mi_web.web_client_actions.search_broken_links()

        .. note::
            |  The function fetches the current URL the driver is on.
            |  Fetches all the 'Anchor links' and 'Images' on the current page and get their responses to find if all the links on the page are working fine or not

        .. warning::
            |  Make sure when you call the function the page itself is working fine and isn't showing any errors

        """
        try:
            # self.web_client_actions.driver.current_url
            total_links = 0
            broken_links = 0
            anchor_links = self.driver.find_elements_by_css_selector("a")
            images = self.driver.find_elements_by_css_selector("img")
            links = anchor_links + images
            self.logger.info("The total number of links on the page are: ", len(links), "\n")
            for link in links:
                flag = 'href'
                # print("Breaking on:", total_links)
                temp_variable = link.get_attribute('href')
                if temp_variable == None:
                    flag = 'src'
                    temp_variable = link.get_attribute('src')
                try:
                    if "https" in temp_variable:
                        r = requests.head(link.get_attribute(flag), verify=False)
                        if r.status_code in [400, 403, 404, 408, 409, 500, 501, 502, 503, 504]:
                            broken_links = broken_links + 1
                            self.logger.info("Broken link: ", link.get_attribute(flag))
                except Exception as e:
                    self.logger.exception(e)
                total_links = total_links + 1
            self.logger.info("\n\nFinal Value:\n", broken_links, " out of ", total_links, " are broken on this URL: ",
                             self.driver.current_url)
            return broken_links

        except Exception as e:
            self.logger.exception("Exception Occurred: " + str(e))

                    pstr_col_no) is str:  # to Fetch all the columns and specific rows
                df_webtable_data = pdf_webtable_df.head(pstr_row_no)

                    pstr_row_no) is list:  # If type is both List , fetch specific rows and columns

                    df_webtable_data = pdf_webtable_df.loc[pstr_row_no, pstr_col_no]

                    df_webtable_data = pdf_webtable_df.iloc[pstr_row_no, pstr_col_no]

                            df_webtable_data = pdf_webtable_df.loc[:, str_col_string1:str_col_string2]
                        else:
                            df_webtable_data = pdf_webtable_df.iloc[:, str_col_string1:str_col_string2]

                    if str_row_string2 is None:

                    else:
                        df_webtable_data = pdf_webtable_df.iloc[str_row_string1:str_row_string2, :]

                        df_webtable_data = pdf_webtable_df.loc[str_row_string1:str_row_string2, str_col_string1]
                    else:
                        df_webtable_data = pdf_webtable_df.iloc[str_row_string1:str_row_string2, str_col_string1]
                else:
                    if type(str_col_string1) is str:
                        df_webtable_data = pdf_webtable_df.loc[str_row_string1:str_row_string2,
                                           str_col_string1:str_col_string2]
                    else:
                        df_webtable_data = pdf_webtable_df.iloc[str_row_string1:str_row_string2,
                                           str_col_string1:str_col_string2]

    def webtable_header_into_list(self, pstr_header_locator):
    """
    Description:
        |  This method reads the headers from web table and return them as list
    :param pstr_header_locator: header locator
    :type pstr_row_locator: string

    :return: list
    Examples:
        |  webtable_header_into_list(pstr_header_locator="xpath=/html/body/table/tbody/tr/th")
    """
    try:
        lst_header_data = []
        for int_header_cnt in range(len(
                self.get_web_elements(pstr_header_locator))):  # fetch the header details and store into List
            str_heder_full_locator = pstr_header_locator + '[' + str(
                int_header_cnt + 1) + ']'  # Create locator for Columns
            str_header_data = self.get_web_element(str_heder_full_locator).text
            lst_header_data.append(str_header_data)  # Load the data into list
        return lst_header_data
    except Exception as e:
        custom_exception_message = "Exception in the : webtable_header_into_list : Method and error is : " + str(
            repr(e))
        self.__exceptions_generic.raise_generic_exception(pstr_message=str(custom_exception_message),
                                                          pbool_insert_kibana=True,
                                                          pbool_trim_log=True, pbool_log_local_exceptions=True,
                                                          pbool_fail_on_exception=False,
                                                          pbool_use_pytest_format=True)
        raise
