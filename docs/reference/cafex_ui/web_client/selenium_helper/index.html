    class SeleniumByHelpers:
    """ The functions within this class are only meant to help us format XPath in the event that two
        attributes and two values are need to locate an element.
    """

    @staticmethod
    def format_first_argument(tag_name, attribute1, value1, operand1):

        # Special case for irregular HTML elements. (E.g. "TSPAN", "SVG", "TEXT" elements.)
        # These require a special XPath selector.
        if tag_name in (special_tags.lower() for special_tags in SeleniumHelper.SPECIAL_HTML_TAGS) and operand1 == "=":
            first_argument = f".//*[name()='{tag_name}'][{attribute1}()='{value1}']"
        elif tag_name in (special_tags.lower() for special_tags in
                          SeleniumHelper.SPECIAL_HTML_TAGS) and operand1 == "*=":
            first_argument = f".//*[name()='{tag_name}'][contains({attribute1}(),'{value1}')]"

        # Matching by "Text" (or "PartialText") requires a specialized XPath selector.
        elif ((attribute1 == "text" or attribute1 == "partialtext") and (operand1 == "=")):
            first_argument = f".//{tag_name}[{attribute1}()=\"{value1}\"]"
        elif ((attribute1 == "text" or attribute1 == "partialtext") and (operand1 == "*=")):
            first_argument = f".//{tag_name}[contains({attribute1}(),\"{value1}\")]"

        # All other "non-special cases" can be handled via CSS_SELECTOR.
        elif operand1 == "=":
            first_argument = f".//{tag_name}[@{attribute1}='{value1}']"
        else:
            first_argument = f".//{tag_name}[contains(@{attribute1},'{value1}')]"

        return first_argument

    @staticmethod
    def format_second_argument(attribute_2, value_2, operand2):
        # Matching by "Text" (or "PartialText") requires a specialized XPath selector.
        if ((attribute_2 == "text" or attribute_2 == "partialtext") and (operand2 == "=")):
            second_argument = f"[{attribute_2}()=\"{value_2}\"]"
        elif ((attribute_2 == "text" or attribute_2 == "partialtext") and (operand2 == "*=")):
            second_argument = f"[contains({attribute_2}(),\"{value_2}\")]"
        elif operand2 == "=":
            second_argument = f"[@{attribute_2}='{value_2}']"
        else:
            second_argument = f"[contains(@{attribute_2},'{value_2}')]"

        return second_argument

    class SeleniumHelper(object):
    HYPHENATED_ATTRIBUTES = ["DataBind", "PartialDataBind", "DataPart", "PartialDataPart",
                             "AriaDescribedBy", "PartialAriaDescribedBy", "DataLevel", "PartialDataLevel",
                             "DataSettingTitle", "PartialDataSettingTitle", "DataToggle", "PartialDataToggle",
                             "DataSubType", "PartialDataSubType", "DataValue", "PartialDataValue", "AriaLabelledBy",
                             "PartialAriaLabelledBy", "DataView", "PartialDataView", "DataRole", "PartialDataRole",
                             "DataColKey", ]

    SPECIAL_HTML_TAGS = ["tspan", "svg", "text", ]

    @staticmethod
    def find_by(tag_name: str, attribute: str = None, value: str = None, attribute2: str = None, value2: str = None):
        """
        Description:
            |   This method can be used as a replacement for manually defining your locator.
            |   This method saves you the time of having to properly format your XPATH/CSS_SELECTOR locators.
            |   There are 3 different ways to use this method:
            |       1. Only specify a tag_name. To do this pass only a single parameter.
            |       2. Specify a tag_name, attribute and value. To do this specify 3 parameters in the respective order.
            |       3. Specify a tag_name, first attribute, first value, second attribute, second value. Use this
                        if one attribute/value is not enough to uniquely identify your element.
            |


            :param tag_name: is the HTML tag of the element you are attempting to locate
            :param attribute: is the name of the attribute within the HTML element you are attempting to locate.
                    ***You may prepend 'Partial' to your attribute if you want to only match a
                    portion of the attributes assigned value.
            :param value: is the text assigned to the first attribute you are attempting to locate the HTML element by
            :param attribute2: is the name of the attribute within the HTML element you are attempting to locate.
                    ***You may prepend 'Partial' to your attribute if you want to only match a
                    portion of the attributes assigned value.
            :param value2: is the text assigned to the second attribute you are attempting to locate the HTML element by
            :return: a properly formatted locator that you can pass in to the selenium find_element or find_elements functions

        note::
            |   If you need to partially match an attribute you must prepend your attribute with 'Partial'
            |   (e.g. if you must partially match a div element with class attribute 'snl-', you would specify the following
            |    SeleniumHelper.find_by("div", "partialclass", "snl-"). Casing should not matter.)
            |
            |   If your attribute contains a hyphen '-' you must remove the hyphen when specifying your attribute.
            |   (e.g. If your attribute is 'data-bind' you would specify SeleniumHelper.find_by("div", "databind", "test")

        Examples:
            |   PyGlobals().web_client_actions.get_web_elements(SeleniumHelper.find_by("div","class","loading-bar"))
            |   will return a collection of elements with HTML div tags

            |   PyGlobals().web_client_actions.get_web_elements(SeleniumHelper.find_by("div", "class", "snl-navigation-module")) will return a collection of
            |    elements which have a tag name of div and a class attribute whose value equals snl-navigation-module exactly.
            |
            |   PyGlobals().web_client_actions.get_web_elements(SeleniumHelper.find_by("div", "partialclass", "snl-")) will return a collection of elements
            |    which have a tag name of div and a class attribute that PARTIALLY matches 'snl-'.
            |
            |   PyGlobals().web_client_actions.get_web_elements(SeleniumHelper.find_by("span", "class", "snl-test", "name", "test_element")) will return
            |    a collection of elements which have a tag name of span, a class attribute of 'snl-test', and a name attribute
            |    of 'test_element'
            |
            |   PyGlobals().web_client_actions.get_web_elements(SeleniumHelper.find_by("select", "PartialDataBind", "gotoHome"))
        """
        if attribute is None:
            return SeleniumHelper._find_by_tag_name(tag_name)
        elif attribute2 is None:
            return SeleniumHelper._find_by_single_attribute_value(tag_name, attribute, value)
        else:
            return SeleniumHelper._find_by_two_attribute_value(tag_name, attribute, value, attribute2, value2)

    @staticmethod
    def find_ancestor_by(tag_name: str, attribute: str = None, value: str = None):
        """
        Description:
            |   This method will help you find ancestor elements.
            |   This method can be used as a replacement for manually defining your locator using (By and a value).
            |   This method saves you the time of having to properly format your XPATH/CSS_SELECTOR locators.
            |   There are 2 different ways to use this method:
            |       1. Only specify a tag_name. To do this pass only a single parameter.
            |       2. Specify a tag_name, attribute and value. To do this specify 3 parameters (tag_name, attribute, value).



        note::
            |   If you need to partially match an attribute you must prepend your attribute with 'Partial'
            |   (e.g. if you must partially match a div element with class attribute 'snl-', you would specify the following
            |    SeleniumHelper.find_ancestor_by("div", "partialclass", "snl-"). Casing should not matter.)
            |
            |   If your attribute contains a hyphen '-' you must remove the hyphen when specifying your attribute.
            |   (e.g. If your attribute is 'data-bind' you would specify SeleniumHelper.find_ancestor_by("div", "databind", "test")

        Examples:
            |   PyGlobals().web_client_actions.get_web_elements(SeleniumHelper.find_ancestor_by("div")) will return a collection of elements with HTML div tags
            |
            |   PyGlobals().web_client_actions.get_web_elements(SeleniumHelper.find_ancestor_by("div", "class", "snl-navigation-module")) will return a collection of
            |    elements which have a tag name of div and a class attribute whose value equals snl-navigation-module exactly.
            |
            |   PyGlobals().web_client_actions.get_web_elements(SeleniumHelper.find_ancestor_by("div", "partialclass", "snl-")) will return a collection of elements
            |    which have a tag name of div and a class attribute that PARTIALLY matches 'snl-'.
            |
            |   PyGlobals().web_client_actions.get_web_elements(SeleniumHelper.find_ancestor_by("select", "PartialDataBind", "gotoHome"))
        """
        if attribute is None:
            return SeleniumHelper._find_ancestor_by_tag_name(tag_name)
        else:
            return SeleniumHelper._find_ancestor_by_single_attribute(tag_name, attribute, value)

    @staticmethod
    def _find_ancestor_by_tag_name(tag_name):
        return f"xpath=./ancestor::{tag_name}"

    @staticmethod
    def _find_ancestor_by_single_attribute(tag_name, attribute, value):
        operand = ''
        attribute = attribute.lower()
        tag_name = tag_name.lower()
        # We can use prepend 'Partial' to our attribute to indicate we want to perform a partial match on the value of the attribute
        if "partial" in attribute:
            operand = '*='
            attribute = attribute[7:]
        else:
            operand = '='

        # Common terminology is to use "URL" for a link attribute, although "HREF" is
        # the actual attribute to use.
        if "url" in attribute:
            attribute = 'href'

        # Handle the hyphenated cases (E.g. convert "DataBind" to "data-bind".)
        if attribute in (hyphenated.lower() for hyphenated in SeleniumHelper.HYPHENATED_ATTRIBUTES):
            before_hyphen = attribute[:4]
            post_hyphen = attribute[4:]
            attribute = f"{before_hyphen}-{post_hyphen}"

        # Special case for irregular HTML elements. (E.g. "TSPAN", "SVG", "TEXT" elements.)
        # These require a special XPath selector.
        if tag_name in (special_tags.lower() for special_tags in SeleniumHelper.SPECIAL_HTML_TAGS) and operand == "=":
            return f"xpath=./ancestor::{tag_name}[{attribute}()='{value}']"
        elif tag_name in (special_tags.lower() for special_tags in
                          SeleniumHelper.SPECIAL_HTML_TAGS) and operand == "*=":
            return f"xpath=./ancestor::{tag_name}[contains({attribute}(),'{value}')]"
        # Matching by "Text" (or "PartialText") requires a specialized XPath selector.
        elif ((attribute == "text") and (operand == "=")):
            return f"xpath=./ancestor::{tag_name}[{attribute}()=\"{value}\"]"
        elif (attribute == "text" and operand == "*="):
            return f"xpath=./ancestor::[contains({attribute}(),\"{value}\")]"
        elif (attribute != "text" and operand == "="):
            return f"xpath=./ancestor::{tag_name}[@{attribute}='{value}']"
        elif (attribute != "text" and operand == "*="):
            return f"xpath=./ancestor::{tag_name}[contains(@{attribute},'{value}')]"

    @staticmethod
    def _find_by_tag_name(tag_name: str):
        """ This method can be used as a replacement for manually defining your By (locator).
            Use this when you only want to search for element(s) by the tag name of the HTML element.
            :param tag_name: is the HTML tag of the element you are attempting to locate
            :return: a properly formatted locator that you can pass in to the selenium find_element or find_elements functions
        """
        return f"xpath=.//{tag_name}"

    @staticmethod
    def _find_by_single_attribute_value(tag_name, attribute, value):
        """ This method can be used as a replacement for manually defining your By (locator)
            Use this when you want to search for an element by tag name and a single attribute
            :rtype: By
            :param tag_name: is the HTML tag of the element you are attempting to locate
            :param attribute: is the name of the attribute within the HTML element you are attempting to locate.
                    ***You may prepend 'Partial' to your attribute if you want to only match a
                    portion of the attributes assigned value.
                     e.g. Take a div element as so <div class="multiselect hui-label snl-open">
                     Finding a div with a partial class attribute of "hui-label" would be done like so
                    by("div", "PartialClass", "hui-label")
            :param value: is the text assigned to the attribute you are attempting to locate the HTML element by
            :return: a properly formatted locator that you can pass in to the selenium find_element or find_elements functions
        """
        operand = ''
        attribute = attribute.lower()
        tag_name = tag_name.lower()
        # We can prepend 'Partial' to our attribute to indicate we want to perform a partial match on the value of the attribute
        if "partial" in attribute:
            operand = '*='
            attribute = attribute[7:]
        else:
            operand = '='

        # Common terminology is to use "URL" for a link attribute, although "HREF" is
        # the actual attribute to use.
        if "url" in attribute:
            attribute = 'href'

        # Handle the hyphenated cases (E.g. convert "DataBind" to "data-bind".)
        if attribute in (hyphenated.lower() for hyphenated in SeleniumHelper.HYPHENATED_ATTRIBUTES):
            before_hyphen = attribute[:4]
            post_hyphen = attribute[4:]
            attribute = f"{before_hyphen}-{post_hyphen}"

        # Special case for irregular HTML elements. (E.g. "TSPAN", "SVG", "TEXT" elements.)
        # These require a special XPath selector.
        if tag_name in (special_tags.lower() for special_tags in SeleniumHelper.SPECIAL_HTML_TAGS) and operand == "=":
            return f"xpath=.//*[name()='{tag_name}'][{attribute}()='{value}']"
        elif tag_name in (special_tags.lower() for special_tags in
                          SeleniumHelper.SPECIAL_HTML_TAGS) and operand == "*=":
            return f"xpath=.//*[name()='{tag_name}'][contains({attribute}(),'{value}')]"

        # Matching by "Text" (or "PartialText") requires a specialized XPath selector.
        elif ((attribute == "text" or attribute == "partialtext") and (operand == "=")):
            return f"xpath=.//{tag_name}[{attribute}()=\"{value}\"]"
        elif ((attribute == "text" or attribute == "partialtext") and (operand == "*=")):
            return f"xpath=.//{tag_name}[contains({attribute}(),\"{value}\")]"

        # All other "non-special cases" can be handled via CSS_SELECTOR.
        else:
            return f"CSS_SELECTOR={tag_name}[{attribute}{operand}'{value}']"

    @staticmethod
    def _find_by_two_attribute_value(tag_name, attribute, value, attribute2, value2):
        operand1 = ''
        operand2 = ''
        attribute = attribute.lower()
        tag_name = tag_name.lower()
        # We can prepend 'Partial' to our attribute to indicate we want to perform a partial match on the value of the attribute
        if "partial" in attribute:
            operand1 = '*='
            attribute = attribute[7:]
        else:
            operand1 = '='

        if "partial" in attribute2:
            operand2 = '*='
            attribute2 = attribute2[7:]
        else:
            operand2 = '='

        # Common terminology is to use "URL" for a link attribute, although "HREF" is
        # the actual attribute to use.
        if "url" in attribute:
            attribute = 'href'

        if "url" in attribute2:
            attribute2 = 'href'

        # Handle the hyphenated cases (E.g. convert "DataBind" to "data-bind".)
        if attribute in (hyphenated.lower() for hyphenated in SeleniumHelper.HYPHENATED_ATTRIBUTES):
            before_hyphen = attribute[:4]
            post_hyphen = attribute[4:]
            attribute = f"{before_hyphen}-{post_hyphen}"

        if attribute2 in (hyphenated.lower() for hyphenated in SeleniumHelper.HYPHENATED_ATTRIBUTES):
            before_hyphen = attribute2[:4]
            post_hyphen = attribute2[4:]
            attribute2 = f"{before_hyphen}-{post_hyphen}"

        first_argument = SeleniumByHelpers.format_first_argument(tag_name, attribute, value, operand1)

        # Now for the second attribute/value
        second_argument = SeleniumByHelpers.format_second_argument(attribute2, value2, operand2)

        return f"xpath={first_argument}{second_argument}"

def find_ancestor_by(tag_name: str, attribute: str = None, value: str = None):
    """
    Description:
        |   This method will help you find ancestor elements.
        |   This method can be used as a replacement for manually defining your locator using (By and a value).
        |   This method saves you the time of having to properly format your XPATH/CSS_SELECTOR locators.
        |   There are 2 different ways to use this method:
        |       1. Only specify a tag_name. To do this pass only a single parameter.
        |       2. Specify a tag_name, attribute and value. To do this specify 3 parameters (tag_name, attribute, value).



    note::
        |   If you need to partially match an attribute you must prepend your attribute with 'Partial'
        |   (e.g. if you must partially match a div element with class attribute 'snl-', you would specify the following
        |    SeleniumHelper.find_ancestor_by("div", "partialclass", "snl-"). Casing should not matter.)
        |
        |   If your attribute contains a hyphen '-' you must remove the hyphen when specifying your attribute.
        |   (e.g. If your attribute is 'data-bind' you would specify SeleniumHelper.find_ancestor_by("div", "databind", "test")

    Examples:
        |   PyGlobals().web_client_actions.get_web_elements(SeleniumHelper.find_ancestor_by("div")) will return a collection of elements with HTML div tags
        |
        |   PyGlobals().web_client_actions.get_web_elements(SeleniumHelper.find_ancestor_by("div", "class", "snl-navigation-module")) will return a collection of
        |    elements which have a tag name of div and a class attribute whose value equals snl-navigation-module exactly.
        |
        |   PyGlobals().web_client_actions.get_web_elements(SeleniumHelper.find_ancestor_by("div", "partialclass", "snl-")) will return a collection of elements
        |    which have a tag name of div and a class attribute that PARTIALLY matches 'snl-'.
        |
        |   PyGlobals().web_client_actions.get_web_elements(SeleniumHelper.find_ancestor_by("select", "PartialDataBind", "gotoHome"))
    """
    if attribute is None:
        return SeleniumHelper._find_ancestor_by_tag_name(tag_name)
    else:
        return SeleniumHelper._find_ancestor_by_single_attribute(tag_name, attribute, value)

    @staticmethod
def find_by(tag_name: str, attribute: str = None, value: str = None, attribute2: str = None, value2: str = None):
    """
    Description:
        |   This method can be used as a replacement for manually defining your locator.
        |   This method saves you the time of having to properly format your XPATH/CSS_SELECTOR locators.
        |   There are 3 different ways to use this method:
        |       1. Only specify a tag_name. To do this pass only a single parameter.
        |       2. Specify a tag_name, attribute and value. To do this specify 3 parameters in the respective order.
        |       3. Specify a tag_name, first attribute, first value, second attribute, second value. Use this
                    if one attribute/value is not enough to uniquely identify your element.
        |


        :param tag_name: is the HTML tag of the element you are attempting to locate
        :param attribute: is the name of the attribute within the HTML element you are attempting to locate.
                ***You may prepend 'Partial' to your attribute if you want to only match a
                portion of the attributes assigned value.
        :param value: is the text assigned to the first attribute you are attempting to locate the HTML element by
        :param attribute2: is the name of the attribute within the HTML element you are attempting to locate.
                ***You may prepend 'Partial' to your attribute if you want to only match a
                portion of the attributes assigned value.
        :param value2: is the text assigned to the second attribute you are attempting to locate the HTML element by
        :return: a properly formatted locator that you can pass in to the selenium find_element or find_elements functions

    note::
        |   If you need to partially match an attribute you must prepend your attribute with 'Partial'
        |   (e.g. if you must partially match a div element with class attribute 'snl-', you would specify the following
        |    SeleniumHelper.find_by("div", "partialclass", "snl-"). Casing should not matter.)
        |
        |   If your attribute contains a hyphen '-' you must remove the hyphen when specifying your attribute.
        |   (e.g. If your attribute is 'data-bind' you would specify SeleniumHelper.find_by("div", "databind", "test")

    Examples:
        |   PyGlobals().web_client_actions.get_web_elements(SeleniumHelper.find_by("div","class","loading-bar"))
        |   will return a collection of elements with HTML div tags

        |   PyGlobals().web_client_actions.get_web_elements(SeleniumHelper.find_by("div", "class", "snl-navigation-module")) will return a collection of
        |    elements which have a tag name of div and a class attribute whose value equals snl-navigation-module exactly.
        |
        |   PyGlobals().web_client_actions.get_web_elements(SeleniumHelper.find_by("div", "partialclass", "snl-")) will return a collection of elements
        |    which have a tag name of div and a class attribute that PARTIALLY matches 'snl-'.
        |
        |   PyGlobals().web_client_actions.get_web_elements(SeleniumHelper.find_by("span", "class", "snl-test", "name", "test_element")) will return
        |    a collection of elements which have a tag name of span, a class attribute of 'snl-test', and a name attribute
        |    of 'test_element'
        |
        |   PyGlobals().web_client_actions.get_web_elements(SeleniumHelper.find_by("select", "PartialDataBind", "gotoHome"))
    """
    if attribute is None:
        return SeleniumHelper._find_by_tag_name(tag_name)
    elif attribute2 is None:
        return SeleniumHelper._find_by_single_attribute_value(tag_name, attribute, value)
    else:
        return SeleniumHelper._find_by_two_attribute_value(tag_name, attribute, value, attribute2, value2)
