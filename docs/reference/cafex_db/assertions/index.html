    """
    Description:
    	|  This Class provides methods which can be used to verify and assert the validations. The methods also allows reporting the PASS/FAIL results into Kibana.

    """

    def __init__(self):
        self.logger = LoggerClass().logger

    def kibana_insert_message_on_result(self, pbool_result, pstr_expected_message, pstr_actual_message=None, **kwargs):
        """
        Description:
        	|  This method is for internal usage and to insert into kibana based on the user inputs and it also has the provision to accept actual message in case of pass and actual message on fail

        :param pbool_result: pbool_result is the result that would come from a validation
        :type pbool_result: Boolean
        :param pstr_expected_message: pstr_expected_message  is the the kwarg parameter and it is the expected value of a this and this will be inserted in kibana
        :type pstr_expected_message: String
        :param pstr_actual_message: pstr_actual_message  is the the kwarg parameter and it is the actual value of a this and this will be inserted in kibana
        :type pstr_actual_message: String
        :param pstr_actual_message_on_pass: pstr_actual_message_on_pass is the actual message when a validation is True
        :type pstr_actual_message_on_pass: String
        :param pstr_actual_message_on_fail: pstr_actual_message_on_fail is the actual message when a validation is False
        :type pstr_actual_message_on_fail: String
        :param pobj_api_response: API Response Object
        :type pobj_api_response: Response
        :param pstr_error_code: pstr_error_code is the error category for assertion failure
        :type pstr_error_code: String

        :return: None


        .. warning::
        	|  this method is intended to be used internally and the end user is not advised to use this method.

        """
        try:
            str_actual_message_on_pass = kwargs.get("pstr_actual_message_on_pass")
            str_actual_message_on_fail = kwargs.get("pstr_actual_message_on_fail")
            pbool_execute_after_failure = kwargs.get("pbool_execute_after_failure", False)
            obj_api_response = kwargs.get('pobj_api_response', None)
            str_error_code = kwargs.get('pstr_error_code', CafeConstants.ERROR_CODE_ZERO)
            if (str_actual_message_on_pass is not None and str_actual_message_on_fail is None) or (
                    str_actual_message_on_pass is None and str_actual_message_on_fail is not None):
                raise Exception(
                    "When 'pstr_actual_message_on_pass' would be passed 'pstr_actual_message_on_fail' should also be passed and vice-versa")
            elif str_actual_message_on_pass is None and str_actual_message_on_fail is None:
                if pstr_actual_message is None:
                    raise Exception(
                        "Neither the pstr_actual_message nor the 'pstr_actual_message_on_pass' and 'pstr_actual_message_on_fail' were passed")
                else:
                    str_actual_message_on_pass = str_actual_message_on_fail = pstr_actual_message
            elif str_actual_message_on_pass == str_actual_message_on_fail:
                raise Exception("pstr_actual_message_on_pass and pstr_actual_message_on_fail cannot be the same")

            if pbool_result:
                PyGlobals().kibana_report.insert_step(pstr_expected_message, str_actual_message_on_pass, "Pass",
                                                      pobj_api_response=obj_api_response)
            else:
                PyGlobals().kibana_report.insert_step(pstr_expected_message, str_actual_message_on_fail, "Fail",
                                                      pobj_api_response=obj_api_response,
                                                      pbool_execute_next_step_after_failure=pbool_execute_after_failure,
                                                      pstr_error=str_error_code)
        except Exception as e:
            PyGlobals().kibana_report.insert_step(pstr_expected_message,
                                                  "Issue with parameters passed in assertion. Details:" + str(repr(e)),
                                                  "Fail")
            pytest.fail(str(e))

        def verify_database_connection(self, pobj_dbconnection, pbool_insert_kibana=True, **kwargs):
            """
            Description:
            	|  The verify_database_connection method verifies whether the DB connection is established or not. It does a basic check of whether the object is null or not.
            	|  Default Values:
            	|  1. pstr_expected_message : 'Connection to database is successful'
            	|  2. pstr_actual_message:
            	|  If pass: 'Connection to database is successful'
            	|  If fail: 'Connection to database is not successful'
            	|  3. pbool_insert_kibana : True

            :param pobj_dbconnection: The database connection object
            :type pobj_dbconnection: object
            :param pbool_insert_kibana: Based on this parameter the verify results will be pushed to the Kibana
            :type pbool_insert_kibana: boolean
            :param pstr_expected_message: Custom Expected Value
            :type pstr_expected_message: kwargs: String
            :param pstr_actual_message: Custom Actual Value
            :type pstr_actual_message: kwargs: String
            :param pstr_actual_message_on_pass: pstr_actual_message_on_pass kwarg parameter is the actual message that would insert into kibana when a validation is True
            :type pstr_actual_message_on_pass: String
            :param pstr_actual_message_on_fail: pstr_actual_message_on_fail kwarg parameter is the actual message that would insert into kibana when a validation is False
            :type pstr_actual_message_on_fail: String

            :return: boolean
            Examples:
            	|  1.Verify the db method with kibana insert:
            	|  obj_assert = Assertions()
            	|  With default_messages: obj_assert.Database().verify_database_connection(obj_db_connection)
            	|  With custom_message: obj_assert.Database().verify_database_connection(obj_db_connection,str_expected_message='database connected',pstr_actual_message='database successfully connected')
            	|  With custom_messages: obj_assert.Database().verify_is_db_connected(obj_db_connection,pstr_expected_message='The DB Connection is successful')
            	|
            	|

            .. note::
            	|  One can use the pstr_expected_message and pstr_actual_message kwargs parameter for sending a customized message

            .. warning::
            	|  The pstr_expected_message and pstr_actual_message are applicable only when the pbool_inser_kibana =True

            """
            try:
                if pobj_dbconnection is not None:
                    bool_result = True
                else:
                    bool_result = False
                if pbool_insert_kibana:
                    str_expected_message = kwargs.get('pstr_expected_message', 'Connection to database is successful')

                    if bool_result:
                        str_actual_message = kwargs.get('pstr_actual_message', 'Connection to database is successful')
                    else:
                        str_actual_message = kwargs.get('pstr_actual_message',
                                                        'Connection to database is not successful')
                    str_actual_message_on_pass = kwargs.get('pstr_actual_message_on_pass')
                    str_actual_message_on_fail = kwargs.get('pstr_actual_message_on_fail')
                    str_error_code = CafeConstants.ERROR_CODE_TWO if not bool_result else None
                    self.assertions.kibana_insert_message_on_result(bool_result, str_expected_message,
                                                                    str_actual_message,
                                                                    pstr_actual_message_on_pass=str_actual_message_on_pass,
                                                                    pstr_actual_message_on_fail=str_actual_message_on_fail,
                                                                    pbool_execute_after_failure=True,
                                                                    pstr_error_code=str_error_code)

                return bool_result
            except Exception as e:
                err_msg = "Exception occured during assertion validation in the step --> " + str(
                    pytest.globalDict["step_name"])
                self.logger.exception(err_msg)
                pytest.fail(err_msg, pytrace=False)

        def assert_database_connection(self, pobj_dbconnection, **kwargs):
            """

            	|  If pass: 'Connection to database is successful'

            :param pobj_dbconnection: The database connection object

            :type pstr_expected_message: kwargs: String
            :param pstr_actual_message: Custom Actual Value

            :param pstr_actual_message_on_pass: pstr_actual_message_on_pass kwarg parameter is the actual message that would insert into kibana when a validation is True
            :type pstr_actual_message_on_pass: String
            :param pstr_actual_message_on_fail: pstr_actual_message_on_fail kwarg parameter is the actual message that would insert into kibana when a validation is False

                pytest.fail(assertion_err_msg, pytrace=False)
            except Exception as e:
                err_msg = "Exception occured during assertion validation in the step --> " + str(
                    pytest.globalDict["step_name"])
                self.logger.exception(err_msg)
                pytest.fail(err_msg, pytrace=False)

        def verify_value_exists_in_column(self, plist_resultlist, plist_header, pdata_value, pbool_insert_kibana=True,
                                          **kwargs):
            """
            Description:
            	|  This method verifies whether the data is present in the respective columns

            :param plist_resultlist: resultlist given by the execute_command

            	|  3. obj_assert.Database().verify_value_exists_in_column(resultlist1,['TEStarttime'],'2013-06-24 15:46:27',pbool_insert_kibana=True,pstr_expected-message='data should be present in columns', pstr_actual_message='data is present in columns')

            .. note::
            	|  1. Data can be of int/float/string/date format

                                                          plist_header))

                                                        str(pdata_value) + ' is present in the columns ' + str(
                                                            plist_header))
                    else:
                        str_actual_message = kwargs.get('pstr_actual_message',
                                                        str(pdata_value) + ' is not present in the columns ' + str(
                                                            plist_header))
                    str_actual_message_on_pass = kwargs.get('pstr_actual_message_on_pass')
                    str_actual_message_on_fail = kwargs.get('pstr_actual_message_on_fail')

    def kibana_insert_message_on_result(self, pbool_result, pstr_expected_message, pstr_actual_message=None, **kwargs):
    """
    Description:
    	|  This method is for internal usage and to insert into kibana based on the user inputs and it also has the provision to accept actual message in case of pass and actual message on fail

    :param pbool_result: pbool_result is the result that would come from a validation
    :type pbool_result: Boolean
    :param pstr_expected_message: pstr_expected_message  is the the kwarg parameter and it is the expected value of a this and this will be inserted in kibana
    :type pstr_expected_message: String
    :param pstr_actual_message: pstr_actual_message  is the the kwarg parameter and it is the actual value of a this and this will be inserted in kibana
    :type pstr_actual_message: String
    :param pstr_actual_message_on_pass: pstr_actual_message_on_pass is the actual message when a validation is True
    :type pstr_actual_message_on_pass: String
    :param pstr_actual_message_on_fail: pstr_actual_message_on_fail is the actual message when a validation is False
    :type pstr_actual_message_on_fail: String
    :param pobj_api_response: API Response Object
    :type pobj_api_response: Response
    :param pstr_error_code: pstr_error_code is the error category for assertion failure
    :type pstr_error_code: String

    :return: None


    .. warning::
    	|  this method is intended to be used internally and the end user is not advised to use this method.

    """
    try:
        str_actual_message_on_pass = kwargs.get("pstr_actual_message_on_pass")
        str_actual_message_on_fail = kwargs.get("pstr_actual_message_on_fail")
        pbool_execute_after_failure = kwargs.get("pbool_execute_after_failure", False)
        obj_api_response = kwargs.get('pobj_api_response', None)
        str_error_code = kwargs.get('pstr_error_code', CafeConstants.ERROR_CODE_ZERO)
        if (str_actual_message_on_pass is not None and str_actual_message_on_fail is None) or (
                str_actual_message_on_pass is None and str_actual_message_on_fail is not None):
            raise Exception(
                "When 'pstr_actual_message_on_pass' would be passed 'pstr_actual_message_on_fail' should also be passed and vice-versa")
        elif str_actual_message_on_pass is None and str_actual_message_on_fail is None:
            if pstr_actual_message is None:
                raise Exception(
                    "Neither the pstr_actual_message nor the 'pstr_actual_message_on_pass' and 'pstr_actual_message_on_fail' were passed")
            else:
                str_actual_message_on_pass = str_actual_message_on_fail = pstr_actual_message
        elif str_actual_message_on_pass == str_actual_message_on_fail:
            raise Exception("pstr_actual_message_on_pass and pstr_actual_message_on_fail cannot be the same")

        if pbool_result:
            PyGlobals().kibana_report.insert_step(pstr_expected_message, str_actual_message_on_pass, "Pass",
                                                  pobj_api_response=obj_api_response)
        else:
            PyGlobals().kibana_report.insert_step(pstr_expected_message, str_actual_message_on_fail, "Fail",
                                                  pobj_api_response=obj_api_response,
                                                  pbool_execute_next_step_after_failure=pbool_execute_after_failure,
                                                  pstr_error=str_error_code)
    except Exception as e:
        PyGlobals().kibana_report.insert_step(pstr_expected_message,
                                              "Issue with parameters passed in assertion. Details:" + str(repr(e)),
                                              "Fail")
        pytest.fail(str(e))
