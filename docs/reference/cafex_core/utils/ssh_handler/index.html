    class SSH_Handler:
    """
    Description:
        |  SSH_Handler class contains methods for below operations
        |  1.Establishing the connection
        |  2.Command execution
        |  3.Download file from remote
        |  4.Upload file to remote
        |  5.Close the SSH connection

    """

    def __init__(self):
        print("created a ssh_handler object")
        self.__obj_exception = Exceptions(os.getcwd())
        self.__obj_generic_exception = self.__obj_exception.Generic()
        self.security = Security()

    def establish_SSH_Connection(self, pstr_servername, **kwargs):
        """
        Description:
                |  This method is used for connecting to a remote machine using SSH

        :param pstr_servername: servername or ip address
        :type pstr_servername: string
        :param pstr_username: username
        :type pstr_username: kwarg string
        :param pstr_password: password
        :type pstr_password: kwarg string
        :param ppem_file: filepath of the pem authentication file
        :type ppem_file: kwarg string

        :return: SSH object
        Examples:
                |  1. Using the pem file
                |  SSHClient=ssh_obj.establish_SSH_Connection('ip_address',pstr_username='username',ppem_file='\path\key.pem')
                |  2. Using the password
                |  SSHClient=ssh_obj.establish_SSH_Connection('ip_address',pstr_username='username',pstr_password='password')

        .. warning::
                |  Either password or pem file is allowed

        """
        try:
            str_username = kwargs.get("pstr_username", None)
            str_password = kwargs.get("pstr_password", None)
            pem_file = kwargs.get("ppem_file", None)
            obj_ssh_client = self.security.establish_ssh_connection(
                pstr_servername, str_username, str_password, pem_file
            )

            return obj_ssh_client
        except Exception as e:
            self.__obj_generic_exception.raise_generic_exception(str(e))

    def __ssh_client(
        self,
        client,
        pint_inbuffer,
        pint_outbuffer,
        pint_width=80,
        pint_height=24,
        pint_width_pixels=0,
        pint_height_pixels=0,
        pstr_term="vt100",
        environment=None,
    ):
        try:
            if pint_width is not None:
                shell = client.invoke_shell(
                    pstr_term,
                    pint_width,
                    pint_height,
                    pint_width_pixels,
                    pint_height_pixels,
                    environment,
                )
            else:
                shell = client.invoke_shell()

            if pint_inbuffer is None or pint_inbuffer < 1:
                stdin = shell.makefile("wb")
            else:
                stdin = shell.makefile("wb", pint_inbuffer)

            if pint_outbuffer is None or pint_outbuffer < 1:
                stdout = shell.makefile("r")
            else:
                stdout = shell.makefile("r", pint_outbuffer)

            return shell, stdin, stdout
        except Exception as e:
            self.__obj_generic_exception.raise_generic_exception(str(e))

    def __execute_command(
        self,
        pobj_client,
        pstr_command,
        pbool_main_channel=True,
        pshell=None,
        pstdin=None,
        pstdout=None,
        pint_inbuffer=None,
        pint_outbuffer=None,
        pint_vt_width=None,
    ):
        try:
            if not pbool_main_channel:

                if pint_vt_width is not None:
                    shell = pobj_client.invoke_shell("vt100", pint_vt_width, 24, 0, 0, None)
                else:
                    shell = pobj_client.invoke_shell()

                if pint_inbuffer is None or pint_inbuffer < 1:
                    stdin = shell.makefile("wb")
                else:
                    stdin = shell.makefile("wb", pint_inbuffer)

                if pint_outbuffer is None or pint_outbuffer < 1:
                    stdout = shell.makefile("r")
                else:
                    stdout = shell.makefile("r", pint_outbuffer)
            else:
                shell = pshell
                stdin = pstdin
                stdout = pstdout
            stdin.write(pstr_command + "\n")
            finish = "End of the command execution"
            echo_cmd = "echo {} $?".format(finish)
            stdin.write(echo_cmd + "\n")
            shin = stdin
            stdin.flush()
            list_shout = []
            list_sherr = []
            bool_exit_status = 0
            bool_flag_capture = False

            for str_line in stdout:
                if (
                    len(re.findall("\[.*?\]\$", str_line)) == 1
                    or len(re.findall("\[.*?\]#", str_line)) == 1
                ) and not bool_flag_capture:
                    bool_flag_capture = True
                if bool_flag_capture:
                    if str(str_line).startswith(finish):
                        bool_exit_status = int(str(str_line).rsplit(maxsplit=1)[1])
                        if bool_exit_status:
                            list_sherr = list_shout
                            list_shout = []
                        break
                    else:
                        list_shout.append(
                            re.compile(r"(\x9B|\x1B\[)[0-?]*[ -/]*[@-~]")
                            .sub("", str_line)
                            .replace("\b", "")
                            .replace("\r", "")
                        )
                    if list_shout and echo_cmd in list_shout[-1]:
                        list_shout.pop()
            if list_sherr.__len__() > 0:
                return bool_exit_status, list_sherr
            else:
                return bool_exit_status, list_shout

        except Exception as e:
            self.__obj_generic_exception.raise_generic_exception(str(e))

    def execute(self, pobj_client, plist_command, **kwargs):
        """
        Description:
                |  This method executes the given commands and will return a list of results and also saves the output to a text file if given

        .. note::
                |  Buffer size depends on the available memory but for efficient use of the buffer size specify as much as
                    needed (do not give excessive buffer size for small task).

        :param pobj_client: SSH Object returned by
        :type pobj_client: object
        :param plist_command: a list of commands that needs to executed
        :type plist_command: list
        :param pstr_filepath: Text file path to store the command ouputs
        :type pstr_filepath: kwarg string
        :param pbool_maintain_channel: Default : True
        :type pbool_maintain_channel: kwarg boolean
        :param pint_inbuffer: Size for input buffer
        :type pint_inbuffer: int
        :param pint_outbuffer: Size for output buffer
        :type pint_outbuffer: int
        :param pint_vt_width: width for virtual terminal
        :type pint_vt_width: int

        :return: list
        Examples:
                |  1. with one command
                |  ssh_obj.execute(SSHClient,['sudo su -'])
                |  2. with multiple commands
                |  ssh_obj.execute(SSHClient,['sudo su -','ls','cd folder'])
                |  3. with multiple commands and with out channel maintainance
                |  ssh_obj.execute(SSHClient,['sudo su -','ls','cd folder'],pbool_maintain_channel=False)
                |  4. with multiple commands and save to text file
                |  ssh_obj.execute(SSHClient,['sudo su -','ls','cd folder'],pstr_filepath="\path\output.txt")
                |  5. With input and output buffers
                |  ssh_obj.execute(SSHClient,['sudo su -','ls','cd folder'], pint_inbuffer=1024, pint_outbuffer=1024)
                |  6. With input, output buffers and virtual terminal width
                |  ssh_obj.execute(SSHClient,['sudo su -','ls','cd folder'], pint_inbuffer=1024, pint_outbuffer=1024, pint_vt_width=200)

        """
        try:
            str_filepath = kwargs.get("pstr_filepath", None)
            bool_maintain_channel = kwargs.get("pbool_maintain_channel", True)
            int_inbuffer = kwargs.get("pint_inbuffer", None)
            int_outbuffer = kwargs.get("pint_outbuffer", None)
            int_vt_width = kwargs.get("pint_vt_width", None)

            shell, stdin, stdout = None, None, None
            if str_filepath is not None and not str_filepath.strip().endswith(".txt"):
                raise Exception("only the text file is allowed")
            if pobj_client is None:
                raise Exception(
                    "The ssh pobj_client passed is none, please check the connection object"
                )
            if type(plist_command) is not list:
                raise Exception("plist_command should be of list")
            list_command_output = []

            if bool_maintain_channel:
                shell, stdin, stdout = self.__ssh_client(
                    pobj_client, int_inbuffer, int_outbuffer, int_vt_width
                )

            for str_command in plist_command:
                if bool_maintain_channel:
                    status, list_output = self.__execute_command(
                        pobj_client, str_command, pshell=shell, pstdin=stdin, pstdout=stdout
                    )
                else:

                    status, list_output = self.__execute_command(
                        pobj_client,
                        str_command,
                        pbool_main_channel=bool_maintain_channel,
                        pint_inbuffer=int_inbuffer,
                        pint_outbuffer=int_outbuffer,
                        pint_vt_width=int_vt_width,
                    )

                list_command_output = list_command_output + list_output
            if str_filepath is not None:
                with open(str_filepath.strip(), "w") as writefile:
                    for data in list_command_output:
                        writefile.write(data)
            return list_command_output
        except Exception as e:
            self.__obj_generic_exception.raise_generic_exception(str(e))

    def download_file_from_remote(self, pobj_client, pstr_remote_path, pstr_local_path):
        """
        Description:
                |  This method is used when we need to perform copy a file from a remote location to local location

        :param pobj_client: The object returned by the establish_connection method
        :type pobj_client: object
        :param pstr_remote_path: remote path of the file (must have the filename along with extension)
        :type pstr_remote_path: string
        :param pstr_local_path: local path of the file (must have the filename along with extension)
        :type pstr_local_path: string

        :return: None
        Examples:
                |  download_file_from_remote(ssh_object,"\ssh_machine\samplefile.txt","\local_path\samplefile.txt")

        .. warning::
                |  1.local and remote path must contain the file name along with extension.
                |  2.The path should not contain any spaces

        """
        try:
            if pobj_client is not None:
                obj_ftp_client = pobj_client.open_sftp()
                obj_ftp_client.get(pstr_remote_path, pstr_local_path)
                obj_ftp_client.close()
            else:
                raise Exception(
                    "The pbj_client is none, looks like you are not passing the correct object or check the the establish connection method return value"
                )
        except Exception as e:
            self.__obj_generic_exception.raise_generic_exception(str(e))

    def establish_SSH_Connection(self, pstr_servername, **kwargs):
    """
    Description:
            |  This method is used for connecting to a remote machine using SSH

    :param pstr_servername: servername or ip address
    :type pstr_servername: string
    :param pstr_username: username
    :type pstr_username: kwarg string
    :param pstr_password: password
    :type pstr_password: kwarg string
    :param ppem_file: filepath of the pem authentication file
    :type ppem_file: kwarg string

    :return: SSH object
    Examples:
            |  1. Using the pem file
            |  SSHClient=ssh_obj.establish_SSH_Connection('ip_address',pstr_username='username',ppem_file='\path\key.pem')
            |  2. Using the password
            |  SSHClient=ssh_obj.establish_SSH_Connection('ip_address',pstr_username='username',pstr_password='password')

    .. warning::
            |  Either password or pem file is allowed

    """
    try:
        str_username = kwargs.get("pstr_username", None)
        str_password = kwargs.get("pstr_password", None)
        pem_file = kwargs.get("ppem_file", None)
        obj_ssh_client = self.security.establish_ssh_connection(
            pstr_servername, str_username, str_password, pem_file
        )

        return obj_ssh_client
    except Exception as e:
        self.__obj_generic_exception.raise_generic_exception(str(e))

    def execute(self, pobj_client, plist_command, **kwargs):
    """
    Description:
            |  This method executes the given commands and will return a list of results and also saves the output to a text file if given

    .. note::
            |  Buffer size depends on the available memory but for efficient use of the buffer size specify as much as
                needed (do not give excessive buffer size for small task).

    :param pobj_client: SSH Object returned by
    :type pobj_client: object
    :param plist_command: a list of commands that needs to executed
    :type plist_command: list
    :param pstr_filepath: Text file path to store the command ouputs
    :type pstr_filepath: kwarg string
    :param pbool_maintain_channel: Default : True
    :type pbool_maintain_channel: kwarg boolean
    :param pint_inbuffer: Size for input buffer
    :type pint_inbuffer: int
    :param pint_outbuffer: Size for output buffer
    :type pint_outbuffer: int
    :param pint_vt_width: width for virtual terminal
    :type pint_vt_width: int

    :return: list
    Examples:
            |  1. with one command
            |  ssh_obj.execute(SSHClient,['sudo su -'])
            |  2. with multiple commands
            |  ssh_obj.execute(SSHClient,['sudo su -','ls','cd folder'])
            |  3. with multiple commands and with out channel maintainance
            |  ssh_obj.execute(SSHClient,['sudo su -','ls','cd folder'],pbool_maintain_channel=False)
            |  4. with multiple commands and save to text file
            |  ssh_obj.execute(SSHClient,['sudo su -','ls','cd folder'],pstr_filepath="\path\output.txt")
            |  5. With input and output buffers
            |  ssh_obj.execute(SSHClient,['sudo su -','ls','cd folder'], pint_inbuffer=1024, pint_outbuffer=1024)
            |  6. With input, output buffers and virtual terminal width
            |  ssh_obj.execute(SSHClient,['sudo su -','ls','cd folder'], pint_inbuffer=1024, pint_outbuffer=1024, pint_vt_width=200)

    """
    try:
        str_filepath = kwargs.get("pstr_filepath", None)
        bool_maintain_channel = kwargs.get("pbool_maintain_channel", True)
        int_inbuffer = kwargs.get("pint_inbuffer", None)
        int_outbuffer = kwargs.get("pint_outbuffer", None)
        int_vt_width = kwargs.get("pint_vt_width", None)

        shell, stdin, stdout = None, None, None
        if str_filepath is not None and not str_filepath.strip().endswith(".txt"):
            raise Exception("only the text file is allowed")
        if pobj_client is None:
            raise Exception(
                "The ssh pobj_client passed is none, please check the connection object"
            )
        if type(plist_command) is not list:
            raise Exception("plist_command should be of list")
        list_command_output = []

        if bool_maintain_channel:
            shell, stdin, stdout = self.__ssh_client(
                pobj_client, int_inbuffer, int_outbuffer, int_vt_width
            )

        for str_command in plist_command:
            if bool_maintain_channel:
                status, list_output = self.__execute_command(
                    pobj_client, str_command, pshell=shell, pstdin=stdin, pstdout=stdout
                )
            else:

                status, list_output = self.__execute_command(
                    pobj_client,
                    str_command,
                    pbool_main_channel=bool_maintain_channel,
                    pint_inbuffer=int_inbuffer,
                    pint_outbuffer=int_outbuffer,
                    pint_vt_width=int_vt_width,
                )

            list_command_output = list_command_output + list_output
        if str_filepath is not None:
            with open(str_filepath.strip(), "w") as writefile:
                for data in list_command_output:
                    writefile.write(data)
        return list_command_output
    except Exception as e:
        self.__obj_generic_exception.raise_generic_exception(str(e))

    def upload_file_to_remote(self, pobj_client, pstr_local_path, pstr_remote_path):
    """
    Description:
            |  This method is used when we need to perform copy a file from a local location to remote location

    :param pobj_client: The object returned by the establish_connection method
    :type pobj_client: object
    :param pstr_local_path: local path of the file (must have the filename along with extension)
    :type pstr_local_path: string
    :param pstr_remote_path: remote path of the file (must have the filename along with extension)
    :type pstr_remote_path: string

    :return: None
    Examples:
            |  upload_file_to_remote(ssh_object,"\local_path\samplefile.txt","\ssh_machine\samplefile.txt")

    .. warning::
            |  1.local and remote path must contain the file name along with extension.
            |  2.The path should not contain any spaces

    """
    try:
        if pobj_client is not None:
            obj_ftp_client = pobj_client.open_sftp()
            obj_ftp_client.put(pstr_local_path, pstr_remote_path)
            obj_ftp_client.close()
        else:
            raise Exception(
                "The pbj_client is none, looks like you are not passing the correct object or check the the establish connection method return value"
            )
    except Exception as e:
        self.__obj_generic_exception.raise_generic_exception(str(e))
