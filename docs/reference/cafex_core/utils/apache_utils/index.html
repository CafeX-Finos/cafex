        def get_component_connections(self, pstr_processor_id, pbln_return_list=False):
            """
            Description:
                |  This method is used to get the connections of component/processor

            :param pstr_processor_id: Processor id
            :type pstr_processor_id: String
            :param pbln_return_list: return type identifier
            :type pbln_return_list: Boolean

            :return: boolean, dict- Tuple without parentheses indicating 'execution_result' and
                                connections dictionary object which contains the 'name' as key and 'id' as value/list of values

            Examples:
                |  get_component_connections('f7b65d6c-d290-32db-6432-322928fdb139')
                |  get_component_connections('f7b65d6c-d290-32db-6432-322928fdb139',False)
                |  get_component_connections('f7b65d6c-d290-32db-6432-322928fdb139',True)

            .. note::
                |  If the pbln_return_list is True return type is a dict of having key:[values]
                |  If the pbln_return_list is False return type is a dict of having key:value

            """
            connections_dict = {"source": {}, "destination": {}}
            try:
                execution, processor = self.__get_processor(pstr_processor_id)
                if execution:
                    connections = nipyapi.canvas.get_component_connections(processor)
                    if len(connections) > 0:
                        if pbln_return_list:
                            for connection in connections:
                                if connection.to_dict()["status"]["source_id"] == pstr_processor_id:
                                    arr_temp = connections_dict["source"].get(
                                        connection.to_dict()["status"]["name"], []
                                    )
                                    arr_temp.append(connection.to_dict()["status"]["id"])
                                    connections_dict["source"][
                                        connection.to_dict()["status"]["name"]
                                    ] = arr_temp
                                elif (
                                    connection.to_dict()["status"]["destination_id"]
                                    == pstr_processor_id
                                ):
                                    arr_temp = connections_dict["destination"].get(
                                        connection.to_dict()["status"]["name"], []
                                    )
                                    arr_temp.append(connection.to_dict()["status"]["id"])
                                    connections_dict["destination"][
                                        connection.to_dict()["status"]["name"]
                                    ] = arr_temp
                        else:
                            for connection in connections:
                                if connection.to_dict()["status"]["source_id"] == pstr_processor_id:
                                    connections_dict["source"][
                                        connection.to_dict()["status"]["name"]
                                    ] = connection.to_dict()["status"]["id"]
                                elif (
                                    connection.to_dict()["status"]["destination_id"]
                                    == pstr_processor_id
                                ):
                                    connections_dict["destination"][
                                        connection.to_dict()["status"]["name"]
                                    ] = connection.to_dict()["status"]["id"]
                        PyGlobals().kibana_report.insert_step(
                            "Successfully get component connections: " + str(pstr_processor_id),
                            "Successfully get component connections: " + str(pstr_processor_id),
                            "Pass",
                        )
                        return True, connections_dict
                    PyGlobals().kibana_report.insert_step(
                        "Successfully get component connections: " + str(pstr_processor_id),
                        "Unable to get component connections: " + str(pstr_processor_id),
                        "Fail",
                    )
                    return False, connections_dict
            except Exception as e:
                self.__obj_generic_exception.raise_generic_exception(
                    pstr_message=str(e),

            pint_wait_interval=5,

            Description:
                |  This method will start and stop the given processor

            :type pstr_server_name: String
            :param pstr_processor_id: nifi processor id
            :type pstr_processor_id: String
            :param pint_min_wait_time: min wait time in seconds
            :type pint_min_wait_time: Integer

            :param pint_wait_interval: pool time to check each time in seconds should be greater-than 0
            :type pint_wait_interval: Integer

            :return: Boolean

                |  start_then_stop_processor('https://guardx-fundamentals-nifi.test.doaas.spglobal.com:8443','f7b65d6c-d290-32db-6432-322928fdb139')

                        + str(pint_max_wait_time)
                        + " sec to process",
                        "Fail",
                    )

                self.change_nifi_processor_state(pstr_server_name, pstr_processor_id, "STOPPED")

        Description:

        :type pstr_server_name: String
        :param pstr_processor_id: nifi processor  id
        :type pstr_nifi_id: String
        :return: Boolean'

    def get_component_connections(self, pstr_processor_id, pbln_return_list=False):
        """
        Description:
            |  This method is used to get the connections of component/processor

        :param pstr_processor_id: Processor id
        :type pstr_processor_id: String
        :param pbln_return_list: return type identifier
        :type pbln_return_list: Boolean

        :return: boolean, dict- Tuple without parentheses indicating 'execution_result' and
                            connections dictionary object which contains the 'name' as key and 'id' as value/list of values

        Examples:
            |  get_component_connections('f7b65d6c-d290-32db-6432-322928fdb139')
            |  get_component_connections('f7b65d6c-d290-32db-6432-322928fdb139',False)
            |  get_component_connections('f7b65d6c-d290-32db-6432-322928fdb139',True)

        .. note::
            |  If the pbln_return_list is True return type is a dict of having key:[values]
            |  If the pbln_return_list is False return type is a dict of having key:value

        """
        connections_dict = {"source": {}, "destination": {}}
        try:
            execution, processor = self.__get_processor(pstr_processor_id)
            if execution:
                connections = nipyapi.canvas.get_component_connections(processor)
                if len(connections) > 0:
                    if pbln_return_list:
                        for connection in connections:
                            if connection.to_dict()["status"]["source_id"] == pstr_processor_id:
                                arr_temp = connections_dict["source"].get(
                                    connection.to_dict()["status"]["name"], []
                                )
                                arr_temp.append(connection.to_dict()["status"]["id"])
                                connections_dict["source"][
                                    connection.to_dict()["status"]["name"]
                                ] = arr_temp
                            elif (
                                connection.to_dict()["status"]["destination_id"]
                                == pstr_processor_id
                            ):
                                arr_temp = connections_dict["destination"].get(
                                    connection.to_dict()["status"]["name"], []
                                )
                                arr_temp.append(connection.to_dict()["status"]["id"])
                                connections_dict["destination"][
                                    connection.to_dict()["status"]["name"]
                                ] = arr_temp
                    else:
                        for connection in connections:
                            if connection.to_dict()["status"]["source_id"] == pstr_processor_id:
                                connections_dict["source"][
                                    connection.to_dict()["status"]["name"]
                                ] = connection.to_dict()["status"]["id"]
                            elif (
                                connection.to_dict()["status"]["destination_id"]
                                == pstr_processor_id
                            ):
                                connections_dict["destination"][
                                    connection.to_dict()["status"]["name"]
                                ] = connection.to_dict()["status"]["id"]
                    PyGlobals().kibana_report.insert_step(
                        "Successfully get component connections: " + str(pstr_processor_id),
                        "Successfully get component connections: " + str(pstr_processor_id),
                        "Pass",
                    )
                    return True, connections_dict
                PyGlobals().kibana_report.insert_step(
                    "Successfully get component connections: " + str(pstr_processor_id),
                    "Unable to get component connections: " + str(pstr_processor_id),
                    "Fail",
                )
                return False, connections_dict
        except Exception as e:
            self.__obj_generic_exception.raise_generic_exception(
                pstr_message=str(e),

    def start_then_stop_processor(
    self,
    pstr_server_name,
    pstr_processor_id,
    pint_min_wait_time=0,
    pint_max_wait_time=50,
    pint_wait_interval=5,
):
    """
    Description:
        |  This method will start and stop the given processor

    :param pstr_server_name: NiFi server URL
    :type pstr_server_name: String
    :param pstr_processor_id: nifi processor id
    :type pstr_processor_id: String
    :param pint_min_wait_time: min wait time in seconds
    :type pint_min_wait_time: Integer
    :param pint_max_wait_time: max wait time in seconds
    :type pint_max_wait_time: Integer
    :param pint_wait_interval: pool time to check each time in seconds should be greater-than 0
    :type pint_wait_interval: Integer

    :return: Boolean

    Examples:
        |  start_then_stop_processor('https://guardx-fundamentals-nifi.test.doaas.spglobal.com:8443','f7b65d6c-d290-32db-6432-322928fdb139')
        |  start_then_stop_processor('https://guardx-fundamentals-nifi.test.doaas.spglobal.com:8443','f7b65d6c-d290-32db-6432-322928fdb139',10)
        |  start_then_stop_processor('https://guardx-fundamentals-nifi.test.doaas.spglobal.com:8443','f7b65d6c-d290-32db-6432-322928fdb139',10,80,20)

    """
    try:
        self.change_nifi_processor_state(pstr_server_name, pstr_processor_id, "RUNNING")

        time.sleep(pint_min_wait_time)
        int_count = 0
        if pint_wait_interval == 0:
            PyGlobals().kibana_report.insert_step(
                "Wait interval should not be 0", "Wait interval is 0", "Fail"
            )
            return False
        int_limit = pint_max_wait_time // pint_wait_interval
        while int_count < int_limit:
            time.sleep(pint_wait_interval)
            str_path = "status/aggregate_snapshot/active_thread_count"
            bln_b, str_c = self.__get_processor(pstr_processor_id)
            int_json_counter = self.get_key_path_value(
                json=str_c.to_dict(), keyPath=str_path, keyPathType="absolute"
            )
            int_count = int_count + 1
            if int(int_json_counter) is 0:
                break
        else:
            PyGlobals().kibana_report.insert_step(
                "Processor should not have active threads ",
                "processor is taking more than "
                + str(pint_max_wait_time)
                + " sec to process",
                "Fail",
            )
            return False

        self.change_nifi_processor_state(pstr_server_name, pstr_processor_id, "STOPPED")
        PyGlobals().kibana_report.insert_step(
            "Processor should get started and stopped",
            "Processor successfully started and stopped",
            "Pass",
        )
        return True
    except Exception as e:
        self.__obj_generic_exception.raise_generic_exception(
            pstr_message=str(e),
            pbool_insert_kibana=True,
            pbool_trim_log=True,
            pbool_log_local_exceptions=True,
            pbool_fail_on_exception=False,
            pbool_use_pytest_format=True,
        )
        return False
