    """
    Description:
        |  This Class provides methods to throw custom exceptions and creates exception logs(full and trimmed) in the project directory once the Project execution is complete. Also, this allows inserting FAIL messages into Reporting BD along with custom exception message.

    .. note::
        |  Create an object for this class as obj = Exceptions(os.getcwd())

    """

    def __init__(self, pstr_current_directory):
        """
        Description:
            |  This method is acts as a constructor

        :param pstr_current_directory: pass this as os.getcwd()
        :type pstr_current_path: String

        :return: None
        """
        self.custom_exception_message_prefix = "CAFE Exception-->"
        self.unhandled_exception_message_prefix = "CAFE Unhandled Exception-->"
        self.current_path = pstr_current_directory
        self.obj_error_category = ErrorCategory()

        self.logger.info("inside log_and_raise_exception of exceptions.py")

    def __get_traceback_trim(
        self, pstr_custom_exception_message, plist_traceback, pbool_use_pytest_format=True
    ):
        """
        Description:
                |  When pbool_use_pytest_format is True:
                |  Returns a dictionary that contains keys [timestamp_iso, scenario_name, step_name, exception_message, custom_exception_message]
                |  When pbool_use_pytest_format is False:
                |  Returns a dictionary that contains keys [timestamp_iso, exception_message, custom_exception_message]

        :param pstr_custom_exception_message:
        :type pstr_custom_exception_message: String
        :param plist_traceback: Any traceback list, example - traceback.extract_stack()
        :type plist_traceback: List
        :param pbool_use_pytest_format: Default value is True
        :type pbool_use_pytest_format: Boolean

        :return: Dictionary
        .. note::
                |  exception_message contains trimmed exception list i.e. all the exceptions entries containing sub string as "/features/" or "/site-packages/cafe/"

        """
        int_size = len(plist_traceback)
        list_trace = []
        dict_trace = {}
        for i in range(int_size):
            str_current = str(plist_traceback[i])
            list_valid_substrings = [
                os.sep + "features" + os.sep,
                os.sep + "site-packages" + os.sep + "cafe" + os.sep,
            ]
            if [item for item in list_valid_substrings if (item in str_current)]:
                list_trace.append(str_current)
        dict_trace["timestamp_iso"] = datetime.datetime.now().isoformat()
        if pbool_use_pytest_format:
            dict_trace["scenario_name"] = str(pytest.globalDict["scenario_name"])
            dict_trace["step_name"] = str(pytest.globalDict["step_name"])
        dict_trace["exception_message"] = list_trace
        dict_trace["custom_exception_message"] = pstr_custom_exception_message
        return dict_trace

    def __get_traceback_complete(
        self, pstr_custom_exception_message, plist_traceback, pbool_use_pytest_format=True
    ):
        """
        Description:
                |  When pbool_use_pytest_format is True:
                |  Returns a dictionary that contains keys [timestamp_iso, scenario_name, step_name, exception_message, custom_exception_message]
                |  When pbool_use_pytest_format is False:
                |  Returns a dictionary that contains keys [timestamp_iso, exception_message, custom_exception_message]

        :param pstr_custom_exception_message:
        :type pstr_custom_exception_message: String
        :param plist_traceback: Any traceback list, example - traceback.extract_stack()
        :type plist_traceback: List
        :param pbool_use_pytest_format: Default value is True
        :type pbool_use_pytest_format: Boolean

        :return: Dictionary
        """
        dict_trace = {}
        dict_trace["timestamp_iso"] = datetime.datetime.now().isoformat()
        dict_trace["exception_message"] = plist_traceback
        dict_trace["custom_exception_message"] = pstr_custom_exception_message
        return dict_trace

    def __log_traceback(
        self, pstr_log_type, pstr_log_filepath, pdict_traceback, pbool_use_pytest_format=True
    ):
        """
        Description:
                |  This method creates a log file and pushes all the log messages to the file in the JSON format.
                |  When pbool_use_pytest_format is True:
                |  Log format is '%(timestamp_iso)s %(scenario_name)s %(step_name)s %(exception_message)s %(custom_exception_message)s'
                |  When pbool_use_pytest_format is False:
                |  Log format is '%(timestamp_iso)s  %(exception_message)s %(custom_exception_message)s'

        :param pstr_log_type:
        :type pstr_log_type: String
        :param pstr_log_filepath: File path to save log messages
        :type pstr_log_filepath: String
        :param pdict_traceback: traceback dictionary
        :type pdict_traceback: Dict
        :param pbool_use_pytest_format: Default value is True
        :type pbool_use_pytest_format: Boolean

        :return: None
        """
        log = logging.getLogger(pstr_log_type)
        log.setLevel(logging.INFO)
        if pbool_use_pytest_format:
            formatter = jsonlogger.JsonFormatter(
                "%(timestamp_iso)s %(scenario_name)s %(step_name)s %(exception_message)s %(custom_exception_message)s"
            )
        else:
            pdict_traceback.pop("scenario_name", None)
            pdict_traceback.pop("step_name", None)
            formatter = jsonlogger.JsonFormatter(
                "%(timestamp_iso)s %(exception_message)s %(custom_exception_message)s"
            )
        file_handler = logging.handlers.RotatingFileHandler(
            filename=pstr_log_filepath, maxBytes=2097152, backupCount=5
        )
        file_handler.setFormatter(formatter)
        log.handlers.clear()
        log.addHandler(file_handler)
        log.propagate = False
        log.info(pdict_traceback)
        for handler in log.handlers[:]:
            handler.close()
            log.removeHandler(handler)

    def run_log_actions(
        self,
        pstr_message,
        plist_traceback,
        pbool_insert_kibana=True,
        pbool_trim_log=True,
        pbool_log_local_exceptions=True,
        pbool_fail_on_exception=True,
        pbool_use_pytest_format=True,
        **kwargs,
    ):
        """
        Description:
                |  Default Behavior is:
                |  Creates two log files(exception_log.json and exception_log_trim.json) into the Project directory. Inserts FAIL record into Reporting DB with custom message(pstr_message) and trimmed exception log and also stops the execution of scenario.
                |  When pbool_log_local_exceptions is False:
                |  No exception log files will be created in Project directory.
                |  When pbool_insert_kibana is False:
                |  No FAIL record will be inserted in Reporting DB.
                |  When pbool_insert_kibana is True and pbool_trim_log is False:
                |  Inserts FAIL record into Reporting DB with custom message(pstr_message) and complete(not trimmed) exception log.
            |  When pbool_fail_on_exception is True:
                |  Execution of scenario will be stopped.
            |  When pbool_use_pytest_format is True:

    def __get_complete_stack_trace(self):
        """
        Description: This method returns the complete stacktrace
        :return: Stack Trace
        """
        try:
            if len(sys.exc_info()) > 0:
                try:
                    if len(sys.exc_info()) > 0:
                        temp = ""
                        if traceback.format_list(traceback.extract_stack(limit=1000)):
                            for i in traceback.format_list(traceback.extract_stack(limit=1000)):
                                if (
                                    f"{os.sep}cafe{os.sep}" in i
                                    or f"{os.sep}features{os.sep}" in i
                                    or f"{os.sep}python_client" in i
                                ) and "__get_complete_stack_trace" not in i:
                                    temp += i
                        if str(temp):
                            return (
                                temp
                                + "\n "
                                + "==========================================\n"
                                + str(traceback.format_exc())
                            )
                        return str(traceback.format_exc())
                except Exception as e1:
                    self.logger.exception(
                        "Exception occured in __get_complete_stack_trace ->" + str(e1)
                    )

        except Exception as e:
            self.logger.exception("Error occurec in __get_complete_stack_trace " + str(e))
            raise Exception("Error occurec in __get_complete_stack_trace " + str(e))

    def __get_trim_stack_trace(self):
        """
        Description: This method returns the trimmed version of stacktrace
        :return: Stack Trace
        """
        try:
            if len(sys.exc_info()) > 0:
                tb = sys.exc_info()[2]
                try:
                    while 1:
                        if not tb.tb_next:
                            break
                        tb = tb.tb_next
                    if "e" in sys.exc_info()[2].tb_frame.f_locals:
                        return repr(sys.exc_info()[2].tb_frame.f_locals["e"])
                    elif "exception" in sys.exc_info()[2].tb_frame.f_locals:
                        return repr(sys.exc_info()[2].tb_frame.f_locals["exception"])
                    elif "exp" in sys.exc_info()[2].tb_frame.f_locals:
                        return repr(sys.exc_info()[2].tb_frame.f_locals["exp"])
                    else:
                        return "Exception Occured"
                except Exception as e1:
                    return "Exception Occured"
        except Exception as e:
            self.logger.exception("Exception occured in __get_trim_stack_trace " + str(e))
            raise Exception("Exception occured in __get_trim_stack_trace " + str(e))

    return dict_trace

    def __get_trim_stack_trace(self):
    """
    Description: This method returns the trimmed version of stacktrace
    :return: Stack Trace
    """
    try:
        if len(sys.exc_info()) > 0:
            tb = sys.exc_info()[2]
            try:
                while 1:
                    if not tb.tb_next:
                        break
                    tb = tb.tb_next
                if "e" in sys.exc_info()[2].tb_frame.f_locals:
                    return repr(sys.exc_info()[2].tb_frame.f_locals["e"])
                elif "exception" in sys.exc_info()[2].tb_frame.f_locals:
                    return repr(sys.exc_info()[2].tb_frame.f_locals["exception"])
                elif "exp" in sys.exc_info()[2].tb_frame.f_locals:
                    return repr(sys.exc_info()[2].tb_frame.f_locals["exp"])
                else:
                    return "Exception Occured"
            except Exception as e1:
                return "Exception Occured"
    except Exception as e:
        self.logger.exception("Exception occured in __get_trim_stack_trace " + str(e))
        raise Exception("Exception occured in __get_trim_stack_trace " + str(e))

    Description:
            |  This method creates a log file and pushes all the log messages to the file in the JSON format.
            |  When pbool_use_pytest_format is True:
            |  Log format is '%(timestamp_iso)s %(scenario_name)s %(step_name)s %(exception_message)s %(custom_exception_message)s'
            |  When pbool_use_pytest_format is False:
            |  Log format is '%(timestamp_iso)s  %(exception_message)s %(custom_exception_message)s'

    :type pstr_log_type: String
    :param pstr_log_filepath: File path to save log messages
    :type pstr_log_filepath: String
    :param pdict_traceback: traceback dictionary
    :type pdict_traceback: Dict
    :param pbool_use_pytest_format: Default value is True
    :type pbool_use_pytest_format: Boolean

    :return: None
    """
    log = logging.getLogger(pstr_log_type)
    log.setLevel(logging.INFO)
    if pbool_use_pytest_format:
        formatter = jsonlogger.JsonFormatter(
            "%(timestamp_iso)s %(scenario_name)s %(step_name)s %(exception_message)s %(custom_exception_message)s"

    else:

            "%(timestamp_iso)s %(exception_message)s %(custom_exception_message)s"
        )

        filename=pstr_log_filepath, maxBytes=2097152, backupCount=5
    )
    file_handler.setFormatter(formatter)
    log.handlers.clear()
    log.addHandler(file_handler)
    log.propagate = False
    log.info(pdict_traceback)
    for handler in log.handlers[:]:
        handler.close()
        log.removeHandler(handler)
