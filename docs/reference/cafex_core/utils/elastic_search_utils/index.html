    class ElasticSearchUtils:
    """
    Description:
        |  This class contains methods to query Kibana or Elastic Search through it's API.

    """

    str_endpoint: str = "/_search/?size={item_size}"
    dict_headers = {
        "Content-Type": "application/json",
        "User-Agent": "CafeAutomationQuery",
        "Accept": "*/*",
        "Cache-Control": "no-cache",
        "Accept-Encoding": "gzip, deflate",
        "Connection": "keep-alive",
    }

    KIBANA_MESSAGE_PATH: str = ".Message"
    URL_REGEX: str = (
        r"http[s]?://(?:[a-zA-Z]|[0-9]|[$-_@.&+]|[!*\(\), ]|(?:%[0-9a-fA-F][0-9a-fA-F]))+"
    )
    EXTRACTION_BLOB_REGEX: str = r'ExtractionBlobUri:[\'"]??([^\'" >]+)'
    __exceptions = Exceptions(os.getcwd())
    __exceptions_generic = __exceptions.Generic()
    __exceptions_services = __exceptions.Services()

    def __init__(self):
        self.logger = LoggerClass().logger

    def generate_payload(self, **kwargs):
        """
        Description:
                |  This method generates the payload based on the filter criteria of the user.

        :param pdict_must_parameters: Must (mandatory) conditions that are being used to filter
        :type pdict_must_parameters: dictionary
        :param pdict_any_parameters: Any (match one of ) of the conditions are used to filter the elastic search
        :type pdict_any_parameters: dictionary
        :param pdict_not_parameters: Not (match none) parameters are used to filter items that should not be filtered
        :type pdict_not_parameters: dictionary
        :param pdict_payload: Custom payload that the user wants to use, instead of using they dynamic construction when this parameter is found we ignore other parameters
        :type pdict_payload: dictionary
        :param pjson_payload: Custom payload that the user wants to use, instead of using they dynamic construction when this parameter is found we ignore other parameters
        :type pjson_payload: json
        :param pstr_index: Index we are trying to filter eg: mi-services-* etc
        :type pstr_index: string
        :param pbln_strict_match: Compares the entire string for matching and does not break down the string into tokens when True
        :type pbln_strict_match: boolean

        :return: dictionary
        Examples:
                |  Use pdict_must_parameters  when we want to match all the conditions defined here i.e  pdict_must_parameters={"Application.Name": "DataApiService_int","Application.LogType": "heartbeat"}, so this translates to a condition Application.Name": "DataApiService_int" and "Application.LogType": "heartbeat".
                |  Use pdict_any_parameters when we want to match any of  the conditions defined here i.e pdict_any_parameters={"Application.Name": "MISearchJava","Application.Name": "SearchService"}, so this translates to a condition "Application.Name": "MISearchJava" or "Application.Name": "SearchService".
                |  Use pdict_not_parameters when we want to match none of  the conditions defined here i.e pdict_not_parameters={"Application.LogType": "heartbeat"}, so this translates to a condition not Application.LogType": "heartbeat".
                |  And we can also combine these condition to formulate complex queries i.e pdict_must_parameters={"@timestamp":{"gt": "now-30m"}},pdict_any_parameters={"Application.Name": "MISearchJava","Application.Name": "SearchService"},pdict_not_parameters={"Application.LogType": "heartbeat"}, so this translates to a condition filter all logs between now and 30m between apps "MISearchJava,SearchService" and exclude heartbeat logs.
                |  Now timerange by default is last 15 minutes, which should be more than enough of most use cases but if you want to query for longer duration it should be passed explicitly in the method generate_payload(Application_Name="MapperService_Int",Application_LogType="exception",@timestamp={"gt": "now-30m"}) Now this would query mapper service exception logs for past 30 minutes, we can also generate custom time periods for any absolute time period checks.
                |  If you want to use your custom payloads and use auto generated payload.then use pdict_payload for initialising the payload or pjson_payload for any json string

        .. code-block:: python

            str_payload  = generate_payload(pdict_must_parameters={"Application.Name": "DataApiService_int","Application.LogType": "heartbeat"}) # Check heartbeat for an app

            str_payload = generate_payload(pdict_must_parameters={"Application.Name": Str_AppName,"Message": "Memory Exception"},pbln_strict_match=True) # Creates a payload to query the elastic search utils, uses strict string match here.

        """

        try:
            dict_payload: dict = None
            dict_must_parameters: dict = kwargs.get("pdict_must_parameters", {})
            dict_any_parameters: dict = kwargs.get("pdict_any_parameters", {})
            dict_not_parameters: dict = kwargs.get("pdict_not_parameters", {})
            dict_payload = kwargs.get("pdict_payload")
            str_index: str = kwargs.get("pstr_index", "*")
            json_payload = kwargs.get("pjson_payload")
            bln_strict_match = kwargs.get("pbln_strict_match", False)

            if not dict_payload is None:
                return dict_payload
            if not json_payload is None:
                dict_payload = json.loads(json_payload)
                return dict_payload

            # build must(mandatory parameters)
            dict_payload = {}
            filters = []
            if not "@timestamp" in dict_must_parameters:
                dict_must_parameters["@timestamp"] = {"gt": "now-15m"}
            if (
                not "Application.LogType" in dict_must_parameters
                and not "Application.LogType" in dict_any_parameters
                and not "Application.LogType" in dict_not_parameters
            ):
                dict_must_parameters["Application.LogType"] = "trace"
            filters.append({"range": {"@timestamp": dict_must_parameters.pop("@timestamp")}})
            if not str_index == "*":
                filters.append({"wildcard": {"_index": {"value": str_index}}})
            # generate payload
            payload = self.__condition_builder(dict_must_parameters, {}, "must", bln_strict_match)
            # either condition
            payload = self.__condition_builder(
                dict_any_parameters, payload, "should", bln_strict_match
            )
            # not filter
            payload = self.__condition_builder(
                dict_not_parameters, payload, "must_not", bln_strict_match
            )
            payload["filter"] = filters
            dict_payload["query"] = {"bool": payload}
            return dict_payload
        except Exception as e:
            self.__exceptions_generic.raise_generic_exception(
                pstr_message=str(e),
                pbool_insert_kibana=True,
                pbool_trim_log=True,
                pbool_log_local_exceptions=True,
                pbool_fail_on_exception=True,
                pbool_use_pytest_format=True,
            )

    def __condition_builder(
        self, pdict_parameters, pdict_payload, pstr_condition, pbln_subcondition_strict_match
    ):
        try:
            if not pdict_parameters.__len__() == 0:
                any_filters = []
                for k, v in pdict_parameters.items():
                    if pbln_subcondition_strict_match:
                        any_filters.append({"match_phrase": {str(k): v}})
                    else:
                        any_filters.append({"match": {str(k): v}})
                pdict_payload.update({pstr_condition: any_filters})
                if pstr_condition == "should":
                    pdict_payload.update({"minimum_should_match": 1})
            return pdict_payload
        except Exception as e:
            self.__exceptions_generic.raise_generic_exception(
                pstr_message=str(e),
                pbool_insert_kibana=True,
                pbool_trim_log=True,
                pbool_log_local_exceptions=True,
                pbool_fail_on_exception=True,
                pbool_use_pytest_format=True,
            )

    def run_query(
        self,
        pstr_host_name,
        pdict_payload,
        pint_message_count=100,
        pdict_headers=dict_headers,
        **kwargs,
    ):
        """
        Description:
                |  This method is used to query Kibana using the filter criteria passed through pdict_payload and fetch the appropriate results.

        :param pstr_host_name: Kibana/Elastic server hostname
        :type pstr_host_name: string
        :param pdict_payload: payload which filters the message for elastic search
        :type pdict_payload: dictionary
        :param pint_message_count: number of elements to filter default is 100
        :type pint_message_count: integer
        :param pdict_headers: headers for the kibana search
        :type pdict_headers: dictionary

        """
        try:
            if pstr_host_name == "" or pstr_host_name == None:
                self.__exceptions_generic.raise_generic_exception(
                    pstr_message="Kibana host name is mandatory",
                    pbool_insert_kibana=True,
                    pbool_trim_log=True,
                    pbool_log_local_exceptions=True,
                    pbool_fail_on_exception=True,
                    pbool_use_pytest_format=True,
                )

            str_end_point = self.str_endpoint.format(item_size=pint_message_count)
            self.response = self.call_request(
                pstr_method="GET",
                pstr_url=pstr_host_name + str_end_point,
                pstr_payload=json.dumps(pdict_payload),
                pdict_headers=pdict_headers,
                **kwargs,
            )

        except Exception as e:
            self.__exceptions_generic.raise_generic_exception(
                pstr_message=str(e),
                pbool_insert_kibana=True,
                pbool_trim_log=True,
                pbool_log_local_exceptions=True,
                pbool_fail_on_exception=True,
                pbool_use_pytest_format=True,
            )

    def __retry_request(self, pfunc_condition, **kwargs):
        """
        Description:
                |  Internal method:
                |  This is a retry method which waits for wait condition to be true. Wait candition can be any lanbda function which returns a bool. If the wait condition is false it polls the request again and checks the condition again.

        :param pfunc_condition: retry function
        :type pfunc_condition: func
        :param pint_max_wait_period_sec: max wait period default is 60 sec
        :type pint_max_wait_period_sec: int
        :param pint_polling_interval_sec: polling interval for retry condition default is 15 sec
        :type pint_polling_interval_sec: int

        :return: bool
        """
        int_max_wait_period: int = kwargs.get("pint_max_wait_period_sec", 60)
        int_polling_interval: int = kwargs.get("pint_polling_interval_sec", 15)
        obj_response = kwargs.get("pobj_response", self.response)
        if obj_response is None:
            self.__exceptions_services.raise_null_value(
                pstr_message="response object is not found",
                pbool_insert_kibana=True,
                pbool_trim_log=True,
                pbool_log_local_exceptions=True,
                pbool_fail_on_exception=True,
                pbool_use_pytest_format=True,
            )
        try:
            int_start_time: int = 0
            while int_start_time < int_max_wait_period:
                if pfunc_condition(obj_response, **kwargs):
                    return True
                time.sleep(int_polling_interval)
                self.logger.info(f" Retrying request: {str(obj_response.url)} ")
                obj_response = requests.Session().send(obj_response.request)
                int_start_time += int_polling_interval
            return False
        except Exception as e:
            self.__exceptions_generic.raise_generic_exception(
                pstr_message=str(e),
                pbool_insert_kibana=True,
                pbool_trim_log=True,
                pbool_log_local_exceptions=True,
                pbool_fail_on_exception=True,
                pbool_use_pytest_format=True,
            )

    def __verify_node(self, pobj_json_response, **kwargs):
        """
        Description:
                |  Internal method
                |  Verifies if the node equals the value.

        :param pobj_json_response: Json Response object on which we run our validation
        :type pobj_json_response: object
        :param pstr_nodepath: path of the property in the json object
        :type pstr_nodepath: str
        :param pobj_expected_value: expected value in the json
        :type pobj_expected_value: object

        :return: bool
        """
        str_nodepath = kwargs.get("pstr_nodepath", "")
        expected_value = kwargs.get("pobj_expected_value", "")

        actual_value = self.get_node_value(
            pstr_node_path=str_nodepath, pjson_content=pobj_json_response.content, **kwargs
        )
        if isinstance(actual_value, itertools.chain) | isinstance(
            actual_value, types.GeneratorType
        ):
            for value in actual_value:
                if value == expected_value:
                    return True
            return False
        return actual_value == expected_value

    def __verify_node_contains(self, pobj_json_response, **kwargs):
        """
        Description:
                |  Internal method
                |  Verifies if the node contains the value

        :param pobj_json_response: Json Response object on which we run our validation
        :type pobj_json_response: object
        :param pfun_condition: contains condition can be customized
        :type pfun_condition: function
        :param pstr_nodepath: path of the property in the json object
        :type pstr_nodepath: str
        :param pobj_expected_value: expected value in the json
        :type pobj_expected_value: object

        :return: bool
        """
        str_nodepath = kwargs.get("pstr_nodepath", "")
        expected_value = kwargs.get("pobj_expected_value", "")
        condition = kwargs.get("pfun_condition", lambda x, y: str(x).__contains__(y))

        actual_value = self.get_node_value(
            pstr_node_path=str_nodepath, pjson_content=pobj_json_response.content, **kwargs
        )
        if not isinstance(actual_value, itertools.chain) and not isinstance(
            actual_value, types.GeneratorType
        ):
            return condition(actual_value, expected_value)
        for item in [match for match in actual_value]:
            if condition(item, expected_value):
                return True
        return False
