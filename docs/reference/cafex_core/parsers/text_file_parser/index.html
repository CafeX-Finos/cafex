    """
    Description:
        |  This class contains method to parse a text file of certain formats and fetch data into a pandas dataframe

    """

    def __init__(self):
        """| Initialize the all arquments."""
        self.str_file_path = ""
        self.int_skip_first_n_rows = 0
        self.int_skip_last_n_rows = 0
        self.list_skip_rows_by_line_no = []
        self.list_all_seperators = []
        self.list_headers = []
        self.table_divider_words = {"start": "", "end": ""}
        self.__exceptions = Exceptions(os.getcwd())
        self.__exceptions_generic = self.__exceptions.Generic()
        self.logger = LoggerClass().logger

    def __check_duplicate_headers_in_row(self, pstr_current_row):
        """
        Description:
            |  This method verifies if the text at a line in the text file contains duplicate headers
            |  This is a helper method and is used for internal use only. It is not advised to be used by the end user.

        :param pstr_current_row: Data row
        :type pstr_current_row: String

        :return: Boolean
        """
        try:
            str_all_headers_for_comparision = "".join(self.list_headers).replace(" ", "").strip()
            return any(
                [
                    "".join(pstr_current_row.split(current_sep)).replace(" ", "").strip()
                    == str_all_headers_for_comparision
                    for current_sep in self.list_all_seperators
                ]
            )
        except Exception as e:
            self.__exceptions_generic.raise_generic_exception(
                pstr_message=str(e),
                pbool_insert_kibana=True,
                pbool_trim_log=True,
                pbool_log_local_exceptions=True,
                pbool_fail_on_exception=False,
                pbool_use_pytest_format=True,
            )

    def __get_row_numbers_having_duplicates_headers(self):
        """
        Description:
            |  This method fetches list of line numbers that contains duplicate headers
            |  This is a helper method and is used for internal use only. It is not advised to be used by the end user.

        :return: List
        """
        try:
            with open(self.str_file_path) as file:
                return [
                    line_num
                    for line_num, current_line in enumerate(file, 1)
                    if self.__check_duplicate_headers_in_row(current_line)
                ]
        except Exception as e:
            self.__exceptions_generic.raise_generic_exception(
                pstr_message=str(e),
                pbool_insert_kibana=True,
                pbool_trim_log=True,
                pbool_log_local_exceptions=True,
                pbool_fail_on_exception=False,
                pbool_use_pytest_format=True,
            )

    def __get_rows_to_skip(self):
        """
        Description:
            |  Finds numbers of rows to skip including top n rows
            |  This is a helper method and is used for internal use only. It is not advised to be used by the end user.

        :return: List
        """
        try:
            plist_duplicate_headers = self.__get_row_numbers_having_duplicates_headers()
            return list(
                set(
                    list(range(0, self.int_skip_first_n_rows))
                    + [n - 1 for n in plist_duplicate_headers]
                    + self.list_skip_rows_by_line_no
                )
            )
        except Exception as e:
            self.__exceptions_generic.raise_generic_exception(
                pstr_message=str(e),
                pbool_insert_kibana=True,
                pbool_trim_log=True,
                pbool_log_local_exceptions=True,
                pbool_fail_on_exception=False,
                pbool_use_pytest_format=True,
            )

    def __format_sep_for_pandas(self):
        """
        Description:
            |  Format 'sep' parameters according to pandas format
            |  This is a helper method and is used for internal use only. It is not advised to be used by the end user.

        :return: String
        """
        try:
            dict_replace_with = {" ": "\s+", "|": "\|"}
            list_all_seps_list_new = [
                (
                    dict_replace_with[current_sep]
                    if current_sep in dict_replace_with.keys()
                    else current_sep
                )
                for current_sep in self.list_all_seperators
            ]
            return "|".join(list_all_seps_list_new)
        except Exception as e:
            self.__exceptions_generic.raise_generic_exception(
                pstr_message=str(e),
                pbool_insert_kibana=True,
                pbool_trim_log=True,
                pbool_log_local_exceptions=True,
                pbool_fail_on_exception=False,
                pbool_use_pytest_format=True,
            )

    def __find_seperator_line_number_for_header_and_data(self):
        """
        Description:
            |  Finds row number that separates headers and table type data
            |  This is a helper method and is used for internal use only. It is not advised to be used by the end user.

        :return: Integer or Boolean
        """
        try:
            with open(self.str_file_path) as file:
                for line_num, current_line in enumerate(file, 1):
                    if current_line.strip() == "*" and line_num > 1:
                        return line_num + 2
            return False
        except Exception as e:
            self.__exceptions_generic.raise_generic_exception(
                pstr_message=str(e),
                pbool_insert_kibana=True,
                pbool_trim_log=True,
                pbool_log_local_exceptions=True,
                pbool_fail_on_exception=False,
                pbool_use_pytest_format=True,
            )

    def __get_headers(self):
        """
        Description:
            |  Find headers from unstructured file. First it looks for the word consisting '_' and then lowercase word from row by ignoring conjunctive word.
            |  This is a helper method and is used for internal use only. It is not advised to be used by the end user.

        :return: List
        """
        try:
            list_all_header = ["temp_index"]
            list_keep_rows_to_find_header = list(range(2, self.no_of_lines_to_skip - 3))
            with open(self.str_file_path) as file:
                all_data = list(file)
                for current_row in list_keep_rows_to_find_header:
                    data_string = all_data[current_row].split()
                    binary_list = ["_" in a for a in data_string]
                    if not any(binary_list):
                        binary_list = [
                            a.islower() and a not in ["on", "of", "per"] for a in data_string
                        ]
                        if not any(binary_list):
                            binary_list = [a[0] == "(" for a in data_string]
                    needed_column = data_string[np.where(binary_list)[0][0]]
                    list_all_header.append(needed_column)
            return list_all_header
        except Exception as e:
            self.__exceptions_generic.raise_generic_exception(
                pstr_message=str(e),
                pbool_insert_kibana=True,
                pbool_trim_log=True,
                pbool_log_local_exceptions=True,
                pbool_fail_on_exception=False,
                pbool_use_pytest_format=True,
            )

    def __find_column_count(self):
        """

                pbool_trim_log=True,
                pbool_log_local_exceptions=True,
                pbool_fail_on_exception=False,
                pbool_use_pytest_format=True,
            )

                error_bad_lines=True,

        except Exception as e:
            self.__exceptions_generic.raise_generic_exception(
                pstr_message=str(e),
                pbool_insert_kibana=True,
                pbool_trim_log=True,
                pbool_log_local_exceptions=True,

    def extract_date_from_string(
        self, pstr_string_to_parse, pstr_input_date_format, pstr_output_date_format
    ):
        """
        Description:
            |  Extracts date from string based on the given input date format and converts into output date format.
            |  If there are multiple dates in a string, it fetch only the first one.

        :param pstr_string_to_parse: String consisting date
        :type pstr_string_to_parse: String
        :param pstr_input_date_format: format of given string
        :type pstr_input_date_format: String
        :param pstr_output_date_format: Desired Date format
        :type pstr_output_date_format: String

        :return: Date
        Examples:
            | extract_date_from_string("04/07/2019 (C) FTSE International Limited 2019...","dd/mm/YY","ddmmYY")
        """
        try:
            get_pstr_input_date_format_sorted = sorted(
                set(pstr_input_date_format), key=pstr_input_date_format.index
            )
            get_pstr_input_date_format = [
                current_conjuctor
                for current_conjuctor in get_pstr_input_date_format_sorted
                if current_conjuctor.isalpha()
            ]

            input_date_conjunction = list(
                set(get_pstr_input_date_format_sorted) - set(get_pstr_input_date_format)
            )[0]

            get_output_date_forma_sorted = sorted(
                set(pstr_output_date_format), key=pstr_output_date_format.index
            )
            get_pstr_output_date_format = [
                current_conjuctor
                for current_conjuctor in get_output_date_forma_sorted
                if current_conjuctor.isalpha()
            ]

            output_date_conjuction_raw = list(
                set(get_output_date_forma_sorted) - set(get_pstr_output_date_format)
            )

            output_date_conjuction = (
                output_date_conjuction_raw[0] if len(output_date_conjuction_raw) > 0 else ""
            )

            pstr_input_date_format = "".join(get_pstr_output_date_format)
            match_pstr_input_date_format = input_date_conjunction.join(
                ["%" + current_verb for current_verb in pstr_input_date_format]
            )
            pstr_output_date_format = "".join(get_pstr_input_date_format)
            match_pstr_output_date_format = output_date_conjuction.join(
                ["%" + current_verb for current_verb in pstr_output_date_format]

    def __headers_with_data_chunk(self):
        """
        Description:
            |  This method handles when the sequence of headers in a file is not uniform. Used for handling files with pint_format_type_id=5.
            |  This is a helper method and is used for internal use only. It is not advised to be used by the end user.

        :return: Dataframe
        """
        try:
            with open(self.str_file_path, "r") as f:
                top_headers = dict()
                headers = dict()
                total_headers = set()
                rows = []
                for row in f.read().split("\n"):
                    row = [col.strip() for col in row.split("  ") if col.strip()]
                    if not headers:
                        headers = row
                        top_headers = row
                        total_headers = set(row)
                        continue
                    if self.__get_header_row_intersection(total_headers, row):
                        headers = row
                        for col in row:
                            total_headers.add(col)
                        continue
                    if not row:
                        continue
                    row = dict(zip(headers, row))
                    rows.append(row)
                collection = []
                for row in rows:
                    row = {col: row.get(col, "NA") for col in total_headers}
                    collection.append(row)
                return pd.DataFrame.from_dict(collection)
        except Exception as e:
            self.__exceptions_generic.raise_generic_exception(

    def extract_date_from_string(
    self, pstr_string_to_parse, pstr_input_date_format, pstr_output_date_format

    """
    Description:
        |  Extracts date from string based on the given input date format and converts into output date format.

    :type pstr_input_date_format: String
    :param pstr_output_date_format: Desired Date format
    :type pstr_output_date_format: String

    :return: Date
    Examples:
        | extract_date_from_string("04/07/2019 (C) FTSE International Limited 2019...","dd/mm/YY","ddmmYY")
    """
    try:
        get_pstr_input_date_format_sorted = sorted(
            set(pstr_input_date_format), key=pstr_input_date_format.index
        )
        get_pstr_input_date_format = [
            current_conjuctor
            for current_conjuctor in get_pstr_input_date_format_sorted
            if current_conjuctor.isalpha()
        ]

        input_date_conjunction = list(
            set(get_pstr_input_date_format_sorted) - set(get_pstr_input_date_format)
        )[0]

        get_output_date_forma_sorted = sorted(
            set(pstr_output_date_format), key=pstr_output_date_format.index
        )
        get_pstr_output_date_format = [
            current_conjuctor
            for current_conjuctor in get_output_date_forma_sorted
            if current_conjuctor.isalpha()
        ]

        output_date_conjuction_raw = list(
            set(get_output_date_forma_sorted) - set(get_pstr_output_date_format)
        )

        output_date_conjuction = (
            output_date_conjuction_raw[0] if len(output_date_conjuction_raw) > 0 else ""
        )

        pstr_input_date_format = "".join(get_pstr_output_date_format)
        match_pstr_input_date_format = input_date_conjunction.join(
            ["%" + current_verb for current_verb in pstr_input_date_format]
        )

            ["%" + current_verb for current_verb in pstr_output_date_format]
        )

            "\d{2}" + input_date_conjunction + "\d{2}" + input_date_conjunction + "\d{4}",
            pstr_string_to_parse,
        )
        return (
            datetime.datetime.strptime(match.group(), match_pstr_input_date_format)
            .date()
            .strftime(match_pstr_output_date_format)
