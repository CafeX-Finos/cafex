    """
    Description:
        |  This class contains methods used to parse json

    """

    def __init__(self):
        self.logger = LoggerClass().logger

    def get_value(self, json_dict: dict, key: str) -> Any:
        """Gets the value of a key from the first level of a JSON dictionary.

        Args:
            json_dict (dict): The JSON dictionary to search.
            key (str): The key to retrieve the value for.

        Returns:
            The value associated with the key, or None if not found.

        Raises:
            ValueError: If json_dict is not a dictionary or the key is not found.
        """
        if not isinstance(json_dict, dict):
            raise ValueError("json_dict must be a dictionary.")
        if not key:
            raise ValueError("key cannot be null or empty.")
        try:
            return json_dict[key]
        except KeyError as e:
            self.logger.exception(f"Error in fetching key value: {e}")
            raise ValueError(f"Key '{key}' not in the JSON dictionary.")

    def get_value_of_key_path(self, json_dict: dict, key_path: str, delimiter: str = "/") -> Any:
        """Gets the value of a key from a nested JSON dictionary using the
        provided key path.

        Args:
            json_dict (dict): The JSON dictionary to search.
            key_path (str): The path to the key, using the specified delimiter.
            delimiter (str, optional): The delimiter used in the key path (default: "/").

        Returns:
            Any: The value associated with the key path, or None if not found.

        Raises:
            ValueError: If json_dict is not a dictionary or the key path is not found.

        Examples:
            1. self.get_value_of_key_path("Data/EnergyActivities",data)
            2. By default, the delimiter is '/'
            3. If the delimiter needs to be changed the method can be called like its mentioned below
            4. self.get_value_of_key_path("Data/EnergyActivities",data,";")
        """
        try:
            if not isinstance(json_dict, dict):
                raise ValueError("json_dict must be a dictionary.")
            if not key_path:
                raise ValueError("key_path cannot be null or empty.")
            key_list = key_path.split(delimiter)
            return reduce(operator.getitem, key_list, json_dict)
        except Exception as e:
            self.logger.exception(f"Error in fetching key value using the provided key path: {e}")
            raise e

    def read_json_file(self, file_path: str) -> dict:
        """Reads a JSON file and returns the parsed dictionary.

        Args:
            file_path: The path to the JSON file.

        Returns:
            The parsed JSON dictionary.

        Raises:
            FileNotFoundError: If the JSON file is not found.
            ValueError: If the JSON file contains invalid JSON.
        """
        if not os.path.exists(file_path):
            raise FileNotFoundError(f"JSON file not found: {file_path}")
        try:
            with open(file_path, "r") as json_file:
                return json.load(json_file)
        except Exception as e:
            self.logger.exception(f"Error parsing JSON file: {e}")
            raise e

    def get_value_of_key(self, json_data: Union[str, dict], key: str, nested: bool = False) -> Any:
        """Gets the value of a key from the JSON data, optionally searching
        nested structures.

        Args:
            json_data: The JSON data to parse (either a string or a dictionary).
            key: The key to retrieve the value for.
            nested: Whether to search for the key in nested structures (default: False).

        Returns:
            The value associated with the key, or a list of values if nested is True and multiple keys are found.

        Raises:
            ValueError: If json_data is not a dictionary or the key is not found.

        Examples:
            1. For simple search of value in first level of json
               get_value_of_key(json='Here the variable containing the json data will be passed',key='The key whose value needs to be fetched')
            2. For getting value of a key in a nested json without determining the exact path to the key
               get_value_of_key(json='json value',key='key value',nested='True')
               returns list of values in case of multiple keys
        """
        if not json_data:
            raise ValueError("json_data cannot be null or empty.")
        if not key:
            raise ValueError("key cannot be null or empty.")
        try:
            if isinstance(json_data, str):
                json_data = json.loads(json_data)

            if nested:
                return nested_lookup(key, json_data)
            else:
                return self.get_value(json_data, key)

        except json.JSONDecodeError as e:
            self.logger.exception(f"Error parsing JSON data: {e}")
            raise ValueError("Invalid JSON data provided.") from e
        except KeyError as e:
            self.logger.exception(f"Key not found: {e}")
            raise ValueError(f"Key '{key}' isn't present in the JSON data.") from e

    def get_value_from_key_path(self, **kwargs: Any) -> Any:
        """Extracts the value at the specified key path from the JSON data.

        Kwargs:
            json: The JSON data (string or dictionary).
            keyPath: The path to the key, using the specified delimiter.
            keyPathType: Either "absolute" or "relative"
            delimiter: The delimiter used in the key path (default: "/").
            key: The key to retrieve the value for when using relative key paths.

        Returns:
            The value associated with the key path.

        Raises:
            ValueError: If required arguments are missing, invalid, or the key path is not found.
            json.JSONDecodeError: If the JSON data is invalid.

        Examples:
          1. For search of value in a pre-determined path of key
             get_value_from_key_path(json='json data',keyPath='node1/node2/key',keyPathType='absolute')
          2. When you want value of all the values of the key from a nested json
             get_value_from_key_path(json='json data',keyPath='node1/node2',keyPathType='relative',key='key')
        """
        try:
            if "json" not in kwargs:
                raise ValueError("json argument is required.")
            if "keyPath" not in kwargs:
                raise ValueError("keyPath argument is required.")
            if "keyPathType" not in kwargs:
                raise ValueError("keyPathType argument is required.")

            json_dict = self.get_dict(kwargs.get("json"))
            key_path = kwargs.get("keyPath")
            key_path_type = kwargs.get("keyPathType").lower()
            delimiter = kwargs.get("delimiter", "/")

            if key_path_type == "absolute":
                return self.get_value_of_key_path(json_dict, key_path, delimiter)
            elif key_path_type == "relative":
                if "key" in kwargs:
                    key = kwargs.get("key")
                    return self.get_json_values_by_key_path(
                        json_dict, delimiter, keypath=key_path, key=key
                    )
                else:
                    raise ValueError("key argument is required if key_path_type is relative")
            else:
                raise ValueError("Invalid keyPathType. Use 'absolute' or 'relative'.")
        except (KeyError, json.JSONDecodeError) as e:
            raise ValueError(f"Error extracting value from key path': {e}") from e

    def print_all_key_values(self, pdata: dict) -> None:
        """Prints all key-value pairs in a nested dictionary structure.

        Args:
            pdata: The nested dictionary to print.

        Raises:
            ValueError: If pdata is not a dictionary.
        """
        if not isinstance(pdata, dict):

    def get_json_values_by_key_path(
        self, pdata: str | dict, delimiter: str = "/", **kwargs: Any
    ) -> Any:
        """Extracts values from a JSON dictionary using a key path.

        Args:
            pdata (Union[str, dict]): The JSON data (string or dictionary).
            delimiter (str, optional): The delimiter used in the key path (default: '/').
            **kwargs (Any): Additional keyword arguments:
                - parser: If provided, parses the `pdata` as a JSON string before processing.
                - keypath: The key path to extract values from.
                - key: The specific sub-child key to retrieve the value for when using relative key paths.

        Returns:
            Any: The value(s) associated with the key path.

        Raises:
            ValueError: If `pdata` or `keypath` is null or empty, or if `key` is not found in the nested dictionary.
            json.JSONDecodeError: If the JSON data is invalid.
        """
        try:
            if not pdata:
                raise ValueError("pdata cannot be null")

            if "parser" in kwargs:
                if not isinstance(pdata, dict):
                    pdata = json.loads(pdata)
                key_path = kwargs.get("keypath")
                if delimiter == "/":
                    key_path = key_path.replace("/", ".")
                json_tree = objectpath.Tree(pdata)
                return json_tree.execute("$." + key_path)

            if "keypath" in kwargs:
                key_path = kwargs.get("keypath")
                root_parent = key_path.split(delimiter)
                root = root_parent[0]
                parent = root_parent[1]
                json_tree = objectpath.Tree(pdata[root])
                result_list = list(json_tree.execute("$.." + parent))
                for result in result_list:
                    sub_child = kwargs.get("key")
                    if "key" in kwargs and isinstance(result, dict) and sub_child is not None:
                        return result.get(sub_child)
                    if sub_child is None:
                        return result

            raise ValueError("keyPathType is not passed")

        except Exception as e:
            raise Exception(str(e))

    def compare_json(
        self,
        expected_json: Union[str, dict],
        actual_json: Union[str, dict],
        ignore_keys: Optional[List[str]] = None,
        ignore_extra: bool = False,
    ) -> Union[bool, Tuple[bool, List[str]]]:
        """Compares two JSON structures (strings or dictionaries) Returns True
        if they are equal, or False with a list of mismatches.

        Args:
            expected_json: The expected JSON data (string or dictionary).
            actual_json: The actual JSON data (string or dictionary).
            ignore_keys: A list of keys to ignore during comparison (default: None).
            ignore_extra: Whether to ignore extra keys in actual_json (default: False).

        Returns:
            True if the JSON structures are equal (considering plist_ignore_keys and ignore_extra),
            otherwise a tuple containing False and a list of mismatches.

        Raises:
            ValueError: If the input JSON data is invalid.

        Examples:
            # Comparing two JSON strings
            result = compare_json('{"a": 1, "b": 2}', '{"a": 1, "b": 2}')
            assert result is True

            # Comparing two dictionaries
            result = compare_json({"a": 1, "b": 2}, {"a": 1, "b": 2})
            assert result is True

            # Ignoring specific keys
            result = compare_json('{"a": 1, "b": 2}', '{"a": 1, "b": 3}', ignore_keys=["b"])
            assert result is True

            # Ignoring extra keys
            result = compare_json('{"a": 1}', '{"a": 1, "b": 2}', ignore_extra=True)
            assert result is True

            # Mismatch example
            result = compare_json('{"a": 1}', '{"a": 2}')
            assert result == (False, ["Expected value for a is 1 but found 2"])
        """

        try:
            if not isinstance(ignore_keys, list):
                raise ValueError("The 'ignore_keys' parameter must be a list")
            if not isinstance(ignore_extra, bool):
                raise ValueError("The 'ignore_extra' parameter must be a boolean")

            if isinstance(expected_json, str):
                expected_json = json.loads(expected_json)
            if isinstance(actual_json, str):
                actual_json = json.loads(actual_json)

            error_list = []

            def compare_dicts(
                expected: Dict[str, Any], actual: Dict[str, Any], path: str = ""
            ) -> None:
                for key, expected_value in expected.items():
                    if key not in ignore_keys:
                        actual_value = actual.get(key)
                        if actual_value is None:
                            error_list.append(f"The key: {path}{key} does not exist in actual json")
                        elif isinstance(expected_value, dict):
                            compare_dicts(expected_value, actual_value, f"{path}{key}.")
                        elif isinstance(expected_value, list):
                            if not isinstance(actual_value, list) or len(expected_value) != len(
                                actual_value
                            ):
                                error_list.append(
                                    f"Expected value for {path}{key} is {expected_value} but found {actual_value}"
                                )
                            else:
                                for i, (exp_item, act_item) in enumerate(
                                    zip(expected_value, actual_value)
                                ):
                                    compare_dicts(exp_item, act_item, f"{path}{key}[{i}].")
                        elif expected_value != actual_value:
                            error_list.append(
                                f"Expected value for {path}{key} is {expected_value} but found {actual_value}"
                            )

                if not ignore_extra:
                    for key in set(actual.keys()) - set(expected.keys()):
                        error_list.append(f"Expected json has an extra key: {path}{key}")

            compare_dicts(expected_json, actual_json)

            if error_list:
                return False, error_list
            return True

        except (ValueError, json.JSONDecodeError) as e:
            self.logger.exception("Error in comparing two JSON structures: %s", e)
            raise ValueError(f"Error comparing JSON objects: {e}")

    def level_based_value(self, pstr_json, pstr_key, plevel=0, **kwargs):
        """
        Description:
            | This method takes a json , a key and the level as input

        :type pstr_json: String
        :param pstr_key: param  key value need to be added
        :type pstr_key: String

        :return: lst_values - The method will return all the values found at level for the given key

        Examples:
            1. level : it signifies the depth of a nested json where you want to look for the key
            2. Note : level here is 0 based index , so the key in the outer most curly-braces would be considered to be at level 0

        """

                    pbool_log_local_exceptions=True,

                    pbool_use_pytest_format=True,
                )

            else:
                current_level = 0
            if not type(pstr_json) is dict:
                dict_response = json.loads(pstr_json)
            else:
                dict_response = pstr_json

            lst_values = []

            for k, v in dict_response.items():
                if pstr_key == k and current_level == plevel:
                    lst_values.append(v)
                    return lst_values

                        v, pstr_key, plevel, actual_level=(current_level + 1)
                    )
                    lst_values = lst_values + temp_value
                if type(v) == list:
                    for x in v:
                        if type(x) is dict and plevel >= current_level:
                            temp_value = self.level_based_value(
                                x, pstr_key, plevel, actual_level=(current_level + 1)
                            )
                            lst_values = lst_values + temp_value
            return lst_values

    def compare_json(
    self,
    expected_json: Union[str, dict],
    actual_json: Union[str, dict],
    ignore_keys: Optional[List[str]] = None,
    ignore_extra: bool = False,
) -> Union[bool, Tuple[bool, List[str]]]:
    """Compares two JSON structures (strings or dictionaries) Returns True
    if they are equal, or False with a list of mismatches.

    Args:
        expected_json: The expected JSON data (string or dictionary).
        actual_json: The actual JSON data (string or dictionary).
        ignore_keys: A list of keys to ignore during comparison (default: None).
        ignore_extra: Whether to ignore extra keys in actual_json (default: False).

    Returns:
        True if the JSON structures are equal (considering plist_ignore_keys and ignore_extra),
        otherwise a tuple containing False and a list of mismatches.

    Raises:
        ValueError: If the input JSON data is invalid.

    Examples:
        # Comparing two JSON strings
        result = compare_json('{"a": 1, "b": 2}', '{"a": 1, "b": 2}')
        assert result is True

        # Comparing two dictionaries
        result = compare_json({"a": 1, "b": 2}, {"a": 1, "b": 2})
        assert result is True

        # Ignoring specific keys
        result = compare_json('{"a": 1, "b": 2}', '{"a": 1, "b": 3}', ignore_keys=["b"])
        assert result is True

        # Ignoring extra keys
        result = compare_json('{"a": 1}', '{"a": 1, "b": 2}', ignore_extra=True)
        assert result is True

        # Mismatch example
        result = compare_json('{"a": 1}', '{"a": 2}')
        assert result == (False, ["Expected value for a is 1 but found 2"])
    """

    try:
        if not isinstance(ignore_keys, list):
            raise ValueError("The 'ignore_keys' parameter must be a list")
        if not isinstance(ignore_extra, bool):
            raise ValueError("The 'ignore_extra' parameter must be a boolean")

        if isinstance(expected_json, str):
            expected_json = json.loads(expected_json)
        if isinstance(actual_json, str):
            actual_json = json.loads(actual_json)

        error_list = []

        def compare_dicts(
            expected: Dict[str, Any], actual: Dict[str, Any], path: str = ""
        ) -> None:
            for key, expected_value in expected.items():
                if key not in ignore_keys:
                    actual_value = actual.get(key)
                    if actual_value is None:
                        error_list.append(f"The key: {path}{key} does not exist in actual json")
                    elif isinstance(expected_value, dict):
                        compare_dicts(expected_value, actual_value, f"{path}{key}.")
                    elif isinstance(expected_value, list):
                        if not isinstance(actual_value, list) or len(expected_value) != len(
                            actual_value
                        ):
                            error_list.append(
                                f"Expected value for {path}{key} is {expected_value} but found {actual_value}"
                            )
                        else:
                            for i, (exp_item, act_item) in enumerate(
                                zip(expected_value, actual_value)
                            ):
                                compare_dicts(exp_item, act_item, f"{path}{key}[{i}].")
                    elif expected_value != actual_value:
                        error_list.append(
                            f"Expected value for {path}{key} is {expected_value} but found {actual_value}"
                        )

            if not ignore_extra:
                for key in set(actual.keys()) - set(expected.keys()):
                    error_list.append(f"Expected json has an extra key: {path}{key}")

        compare_dicts(expected_json, actual_json)

        if error_list:
            return False, error_list
        return True

    except (ValueError, json.JSONDecodeError) as e:
        self.logger.exception("Error in comparing two JSON structures: %s", e)
        raise ValueError(f"Error comparing JSON objects: {e}")
